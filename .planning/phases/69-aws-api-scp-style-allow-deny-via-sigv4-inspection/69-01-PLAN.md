```go
// In SourceAccessSpec struct (around line 217):
type SourceAccessSpec struct {
    GitHub *GitHubAccess `yaml:"github,omitempty"`
    // AWS controls AWS API service-level allowlist gating in the http-proxy. Phase 69.
    AWS *AWSAccess `yaml:"aws,omitempty"`
}

// New struct after GitHubAccess:
// AWSAccess defines AWS API service-level allowlist gating.
// Inspection enum: "off" (default; absence ≡ off), "observe" (log-only),
// "enforce" (block non-allowlisted services with HTTP 403).
// Allowlist is required when Inspection != "off". Entries are SigV4 service
// slugs (e.g. "s3", "sts", "bedrock-runtime"). The single-element list
// ["*"] is the only legal wildcard form. Mixing "*" with explicit entries
// is a validation error. Operation-level suffixes (e.g. "s3:GetObject") are
// accepted forward-compatibly but ignored past the first ":" in v1.
type AWSAccess struct {
    Inspection string   `yaml:"inspection,omitempty"`
    Allowlist  []string `yaml:"allowlist,omitempty"`
}
```

1. **Edit `pkg/profile/validate.go`** — append a new block of rules to `ValidateSemantic`. Place it after the Slack inbound block (around line 350+ in the current file). Use the exact pattern from Phase 67's inbound rules.

```go
// Phase 69 — AWS API allowlist validation rules.
if p.Spec.SourceAccess.AWS != nil {
    aws := p.Spec.SourceAccess.AWS
    mode := aws.Inspection
    if mode == "" {
        mode = "off"
    }

    // AW4: enum check.
    switch mode {
    case "off", "observe", "enforce":
        // ok
    default:
        errs = append(errs, ValidationError{
            Path:    "spec.sourceAccess.aws.inspection",
            Message: fmt.Sprintf("inspection must be one of off|observe|enforce; got %q", aws.Inspection),
        })
    }

    if mode != "off" {
        // AW3: allowlist required when not off.
        if len(aws.Allowlist) == 0 {
            errs = append(errs, ValidationError{
                Path:    "spec.sourceAccess.aws.allowlist",
                Message: "allowlist is required when inspection != off; use [\"*\"] for wildcard",
            })
        }

        // AW2: wildcard-mixing.
        hasWildcard := false
        hasExplicit := false
        for _, svc := range aws.Allowlist {
            if svc == "*" {
                hasWildcard = true
            } else if svc != "" {
                hasExplicit = true
            }
        }
        if hasWildcard && hasExplicit {
            errs = append(errs, ValidationError{
                Path:    "spec.sourceAccess.aws.allowlist",
                Message: `["*"] wildcard may not be mixed with explicit service entries; use either ["*"] or an explicit list`,
            })
        }

        // AW1: Bedrock cross-check (only when enforce).
        if mode == "enforce" && p.Spec.Budget != nil && p.Spec.Budget.AI != nil && p.Spec.Budget.AI.MaxSpendUSD > 0 {
            // Wildcard satisfies the cross-check.
            satisfied := hasWildcard
            if !satisfied {
                for _, svc := range aws.Allowlist {
                    slug := strings.SplitN(svc, ":", 2)[0]
                    if slug == "bedrock-runtime" {
                        satisfied = true
                        break
                    }
                }
            }
            if !satisfied {
                errs = append(errs, ValidationError{
                    Path:    "spec.sourceAccess.aws.allowlist",
                    Message: "inspection: enforce with a non-zero AI budget requires bedrock-runtime in the allowlist — budget enforces metering on bedrock-runtime; add bedrock-runtime or set spec.budget.ai.maxSpendUSD to zero",
                })
            }
        }
    }
}
```
