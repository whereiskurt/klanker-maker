```go
// Package aws — vetted list of common SigV4 service slugs.
//
// Used by km doctor's aws_allowlist_known_services check (Phase 69) to WARN
// (not FAIL) on profile entries that are not in this list. WARN-not-FAIL is
// intentional: AWS occasionally renames services or launches new ones, and
// blocking a profile because the platform's vetted list is stale would be a
// poor operator experience.
//
// UPDATE PROCESS:
//
//   1. Encounter an unknown service slug in a real SigV4 Authorization header
//      (visible in proxy logs as event_type=aws_api_unsigned with a hint, or
//      via aws_allowlist_known_services WARN output).
//   2. Confirm the slug is real by cross-referencing
//      https://docs.aws.amazon.com/general/latest/gr/aws-service-information.html
//      or by inspecting the AWS SDK service ID (e.g., the Go SDK's service
//      package name normalized to its endpoint slug).
//   3. Add the slug to KnownSigV4Services below, keeping the slice
//      alphabetically sorted. Run `go test ./pkg/aws/... -run TestKnownSigV4`.
//   4. Open a PR. The list intentionally lives in version control rather than
//      a remote feed so the platform's "known services" set is auditable.
//
// The list is conservative — it covers services likely to appear in real
// sandbox workloads. It is NOT exhaustive (AWS has 200+ services).
package aws

import (
    "strings"
    "sync"
)

// KnownSigV4Services is the curated list of common SigV4 service slugs.
// Sorted alphabetically. Case-sensitive — matches the SigV4 wire format.
//
//nolint:gochecknoglobals // intentional package-level data table
var KnownSigV4Services = []string{
    "acm",
    "apigateway",
    "appsync",
    "athena",
    "bedrock",
    "bedrock-runtime",
    "cloudformation",
    "cloudfront",
    "cloudtrail",
    "cloudwatch",
    "codebuild",
    "codecommit",
    "cognito-identity",
    "cognito-idp",
    "dynamodb",
    "ec2",
    "ecr",
    "ecs",
    "eks",
    "elasticache",
    "elasticfilesystem",
    "elasticloadbalancing",
    "es", // OpenSearch / Elasticsearch
    "events",
    "glue",
    "iam",
    "kinesis",
    "kms",
    "lambda",
    "logs", // CloudWatch Logs
    "monitoring", // CloudWatch metrics (PutMetricData)
    "rds",
    "redshift",
    "route53",
    "s3",
    "sagemaker",
    "secretsmanager",
    "ses",
    "sns",
    "sqs",
    "ssm",
    "states", // Step Functions
    "sts",
    "transcribe",
    "translate",
    "waf",
    "wafv2",
}

var (
    knownSetOnce sync.Once
    knownSet     map[string]struct{}
)

func buildKnownSet() {
    knownSet = make(map[string]struct{}, len(KnownSigV4Services))
    for _, s := range KnownSigV4Services {
        knownSet[s] = struct{}{}
    }
}

// IsKnownSigV4Service reports whether slug appears in KnownSigV4Services.
// Case-sensitive. Strips operation-level suffix (e.g., "s3:GetObject" → "s3")
// before lookup so forward-compatible profile entries work.
func IsKnownSigV4Service(slug string) bool {
    if slug == "" {
        return false
    }
    knownSetOnce.Do(buildKnownSet)
    // Strip past first colon (forward-compat for v2 operation-level entries).
    if i := strings.Index(slug, ":"); i >= 0 {
        slug = slug[:i]
    }
    _, ok := knownSet[slug]
    return ok
}
```

```go
package aws

import (
    "sort"
    "testing"
)

func TestKnownSigV4Services_Sorted(t *testing.T) {
    if !sort.StringsAreSorted(KnownSigV4Services) {
        t.Fatalf("KnownSigV4Services must be sorted alphabetically")
    }
}

func TestKnownSigV4Services_NoDuplicates(t *testing.T) {
    seen := make(map[string]bool)
    for _, s := range KnownSigV4Services {
        if seen[s] {
            t.Errorf("duplicate entry: %s", s)
        }
        seen[s] = true
    }
}

func TestIsKnownSigV4Service(t *testing.T) {
    cases := []struct {
        slug string
        want bool
    }{
        {"s3", true},
        {"sts", true},
        {"dynamodb", true},
        {"bedrock-runtime", true},
        {"logs", true},
        {"events", true},
        {"sqs", true},
        // Operation-level forward-compat (colon-stripping):
        {"s3:GetObject", true},
        {"bedrock-runtime:InvokeModel", true},
        // Unknown:
        {"aws-fictional-service-xyz", false},
        {"", false},
        // Case-sensitive:
        {"S3", false},
        {"BEDROCK-RUNTIME", false},
    }
    for _, tc := range cases {
        t.Run(tc.slug, func(t *testing.T) {
            got := IsKnownSigV4Service(tc.slug)
            if got != tc.want {
                t.Errorf("IsKnownSigV4Service(%q) = %v; want %v", tc.slug, got, tc.want)
            }
        })
    }
}
```
