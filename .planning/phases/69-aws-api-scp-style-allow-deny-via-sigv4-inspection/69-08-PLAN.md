const (
    awsInspectionUIDCheckName    = "aws_inspection_uid_map"
    awsAllowlistKnownCheckName   = "aws_allowlist_known_services"
    awsInspectionPlatformUIDMax  = 1000 // mirror compiler default; could be plumbed from SSM/config

// inspectionSandbox is the minimal projection needed by the doctor checks.
// Built by listInspectionEnabledSandboxes, which reads SandboxMetadata + the
// full profile body from S3 (SandboxMetadata alone does not carry inspection state).
type inspectionSandbox struct {
    ID         string
    InstanceID string  // EC2 instance id (only populated for EC2 substrate)
    Region     string
    Inspection string   // "observe" | "enforce" (off-mode sandboxes are filtered out)
    Allowlist  []string
}

// listInspectionEnabledSandboxes returns sandboxes whose profiles set
// spec.sourceAccess.aws.inspection != off.
//
// Important: SandboxMetadata in DDB/S3 does NOT carry inspection state directly.
// We must fetch the profile YAML from S3 for each candidate sandbox and parse
// it. This is acceptable scope for doctor (read-mostly, periodic sweep). If S3
// fetch or YAML parse fails for a given sandbox, log a debug-level warning and
// skip that sandbox — do not fail the whole listing.
//
// Inputs:
//   - lister: typically a closure over kmaws.ListAllSandboxMetadataDynamo
//   - s3Client: for downloadProfileFromS3
//   - artifactsBucket: the state bucket name (from km-config)
func listInspectionEnabledSandboxes(
    ctx context.Context,
    lister func(ctx context.Context) ([]kmaws.SandboxMetadata, error),
    s3Client *s3.Client,
    artifactsBucket string,
) ([]inspectionSandbox, error) {
    all, err := lister(ctx)
    if err != nil {
        return nil, fmt.Errorf("list sandboxes: %w", err)
    }
    var out []inspectionSandbox
    for _, sb := range all {
        body, err := downloadProfileFromS3(ctx, s3Client, artifactsBucket, sb.SandboxID)
        if err != nil {
            log.Debug().Str("sandbox_id", sb.SandboxID).Err(err).Msg("skip sandbox: profile fetch failed")
            continue
        }
        p, err := profile.UnmarshalYAML(body) // use the existing public unmarshal helper from pkg/profile
        if err != nil {
            log.Debug().Str("sandbox_id", sb.SandboxID).Err(err).Msg("skip sandbox: profile parse failed")
            continue
        }
        if p.Spec.SourceAccess.AWS == nil {
            continue
        }
        mode := p.Spec.SourceAccess.AWS.Inspection
        if mode == "" || mode == "off" {
            continue
        }
        // EC2 substrate exposes InstanceID via metadata.json (read by km list/status).
        // The exact field is whatever the existing km status reads — confirm via
        // grep for "InstanceID" against pkg/aws/metadata.go and use the same field.
        instanceID := "" // populate from the fuller metadata record if EC2; leave empty for Docker
        out = append(out, inspectionSandbox{
            ID:         sb.SandboxID,
            InstanceID: instanceID,
            Region:     sb.Region,
            Inspection: mode,
            Allowlist:  p.Spec.SourceAccess.AWS.Allowlist,
        })
    }
    return out, nil
}

// checkAWSInspectionUIDMap probes a sample inspection-enabled sandbox to verify
// the km-mail-poller system user exists and has uid < KM_AWS_PLATFORM_UID_MAX.
// Phase 69 SC-6a.
func checkAWSInspectionUIDMap(ctx context.Context, ssmClient SSMRunCommandAPI, sandboxes []inspectionSandbox) CheckResult {
    name := awsInspectionUIDCheckName
    if len(sandboxes) == 0 {
        return CheckResult{Name: name, Status: CheckSkipped, Message: "no inspection-enabled sandboxes in this region"}
    }

    // Pick the first running sandbox; if its check fails for transient reasons
    // (SSM not reachable), report ERROR — the doctor.go wrapper will demote to WARN.
    sb := sandboxes[0]
    out, err := runSSMCommand(ctx, ssmClient, sb.InstanceID, fmt.Sprintf("id -u %s", awsInspectionProbeUser))
    if err != nil {
        return CheckResult{
            Name:    name,
            Status:  CheckError,
            Message: fmt.Sprintf("SSM RunCommand `id -u %s` failed on sandbox %s: %v", awsInspectionProbeUser, sb.ID, err),
        }
    }
    out = strings.TrimSpace(out)
    if out == "" {
        return CheckResult{
            Name:        name,
            Status:      CheckWarn,
            Message:     fmt.Sprintf("sandbox %s: user %s not found in /etc/passwd", sb.ID, awsInspectionProbeUser),
            Remediation: "Check userdata template — km-mail-poller should be created during cloud-init",
        }
    }
    uid, err := strconv.ParseUint(out, 10, 32)
    if err != nil {
        return CheckResult{
            Name:    name,
            Status:  CheckError,
            Message: fmt.Sprintf("sandbox %s: cannot parse uid output %q: %v", sb.ID, out, err),
        }
    }
    if uid >= awsInspectionPlatformUIDMax {
        return CheckResult{
            Name:        name,
            Status:      CheckError,
            Message:     fmt.Sprintf("sandbox %s: %s uid=%d is NOT below KM_AWS_PLATFORM_UID_MAX=%d — platform calls would NOT bypass the AWS gate!", sb.ID, awsInspectionProbeUser, uid, awsInspectionPlatformUIDMax),
            Remediation: "Adjust userdata to create km-mail-poller with a system uid (<1000), or raise KM_AWS_PLATFORM_UID_MAX",
        }
    }
    return CheckResult{
        Name:    name,
        Status:  CheckOK,
        Message: fmt.Sprintf("sandbox %s: %s uid=%d (below threshold %d)", sb.ID, awsInspectionProbeUser, uid, awsInspectionPlatformUIDMax),
    }
}

// checkAWSAllowlistKnownServices verifies every allowlist entry against the
// vetted SigV4 service list. WARN (not FAIL) on unknowns. Phase 69 SC-6b.
func checkAWSAllowlistKnownServices(ctx context.Context, sandboxes []inspectionSandbox) CheckResult {
    name := awsAllowlistKnownCheckName
    if len(sandboxes) == 0 {
        return CheckResult{Name: name, Status: CheckSkipped, Message: "no inspection-enabled sandboxes in this region"}
    }

    type unknown struct {
        Sandbox string
        Slug    string
    }
    var unknowns []unknown
    total := 0
    for _, sb := range sandboxes {
        for _, entry := range sb.Allowlist {
            if entry == "*" {
                continue
            }
            total++
            if !kmaws.IsKnownSigV4Service(entry) {
                unknowns = append(unknowns, unknown{Sandbox: sb.ID, Slug: entry})
            }
        }
    }

    if len(unknowns) == 0 {
        return CheckResult{
            Name:    name,
            Status:  CheckOK,
            Message: fmt.Sprintf("all %d allowlist entries across %d sandboxes are known SigV4 services", total, len(sandboxes)),
        }
    }

    msgs := make([]string, 0, len(unknowns))
    for _, u := range unknowns {
        msgs = append(msgs, fmt.Sprintf("%s: %s", u.Sandbox, u.Slug))
    }
    return CheckResult{
        Name:        name,
        Status:      CheckWarn,
        Message:     fmt.Sprintf("%d unknown service slug(s): %s", len(unknowns), strings.Join(msgs, ", ")),
        Remediation: "If these slugs are real but new AWS services, add them to pkg/aws/sigv4_services.go::KnownSigV4Services and submit a PR. Otherwise, fix the typos in the affected profiles.",
    }
}
