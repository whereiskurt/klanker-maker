// Package httpproxy — AWS API allowlist gate. Phase 69.
//
// Inspects SigV4-signed traffic to *.amazonaws.com (and partition variants),
// matches the service slug against an operator-declared allowlist, and either
// allows, blocks, or merely logs the call depending on the inspection mode.
// Platform-system users (uid < KM_AWS_PLATFORM_UID_MAX) bypass the gate and
// emit aws_api_platform events.
//
// Out of scope (deferred): operation-level entries (s3:GetObject), region
// pinning, IMDS gating, VPC endpoints. Schema accepts colon-suffixed entries
// forward-compatibly but ignores past-colon content in v1.
package httpproxy

import (
    "encoding/json"
    "net/http"
    "os/user"
    "regexp"
    "strconv"
    "strings"
    "sync"

    "github.com/elazarl/goproxy"
    "github.com/rs/zerolog/log"
)

// awsHostRegex matches AWS API endpoints across all partitions (locked decision, 69-CONTEXT.md).
var awsHostRegex = regexp.MustCompile(`(?i)(.*\.)?(amazonaws\.com|amazonaws\.com\.cn|amazonaws-us-gov\.com|api\.aws)$`)

// awsInspectionOptions is the runtime config for the AWS allowlist gate.
type awsInspectionOptions struct {
    mode      string   // "off" | "observe" | "enforce"
    allowlist []string // service slugs or ["*"]
}

// WithAWSAllowlist enables SigV4-based AWS API service gating.
// mode must be "off", "observe", or "enforce". Empty string is treated as "off".
// When mode is "off", no handlers are registered.
func WithAWSAllowlist(mode string, allowlist []string) ProxyOption {
    return func(_ *goproxy.ProxyHttpServer, cfg *proxyConfig) {
        cfg.awsInspection = mode
        cfg.awsAllowlist = allowlist
    }
}

// extractSigV4Service parses the SigV4 credential scope from a request.
// Returns the service slug, region, and a found boolean.
// Supports AWS4-HMAC-SHA256 + AWS4-ECDSA-P256-SHA256 (sigv4a) Authorization headers
// and X-Amz-Credential query-parameter fallback for pre-signed URLs.
func extractSigV4Service(req *http.Request) (service, region string, found bool) {
    auth := req.Header.Get("Authorization")
    if strings.HasPrefix(auth, "AWS4-") {
        // Authorization: AWS4-HMAC-SHA256 Credential=AKIA.../20260504/us-east-1/s3/aws4_request, SignedHeaders=..., Signature=...
        for _, part := range strings.Split(auth, ",") {
            part = strings.TrimSpace(part)
            // Some headers omit the algorithm prefix's space (rare); split on first space inside the part too.
            if idx := strings.Index(part, "Credential="); idx >= 0 {
                cred := part[idx+len("Credential="):]
                if svc, rg, ok := parseCredentialScope(cred); ok {
                    return svc, rg, true
                }
            }
        }
    }
    if cred := req.URL.Query().Get("X-Amz-Credential"); cred != "" {
        if svc, rg, ok := parseCredentialScope(cred); ok {
            return svc, rg, true
        }
    }
    return "", "", false
}

// parseCredentialScope splits "AKIA.../<date>/<region>/<service>/aws4_request" → service, region.
func parseCredentialScope(cred string) (service, region string, ok bool) {
    parts := strings.SplitN(cred, "/", 6)
    if len(parts) < 5 {
        return "", "", false
    }
    // parts: [accessKeyID, date, region, service, "aws4_request", ...]
    return parts[3], parts[2], true
}

// matchAllowlist returns true when service is in allowlist (with ["*"] wildcard support
// and colon-suffix stripping for forward-compatible operation-level entries).
func matchAllowlist(allowlist []string, service string) bool {
    for _, entry := range allowlist {
        if entry == "*" {
            return true
        }
        slug := strings.SplitN(entry, ":", 2)[0]
        if slug == service {
            return true
        }
    }
    return false
}

// awsBlockedBody is the JSON response body for 403 responses from the AWS gate.
type awsBlockedBody struct {
    Error   string `json:"error"`   // always "KM_AWS_BLOCKED"
    Service string `json:"service"` // SigV4 service slug
    Reason  string `json:"reason"`  // "not_in_allowlist" | "empty_allowlist"
}

// AWSBlockedResponse builds a 403 Forbidden response for blocked AWS API calls.
// The response body is JSON with error=KM_AWS_BLOCKED, service, reason — the AWS CLI
// surfaces the body in stderr so operators can identify the cause.
func AWSBlockedResponse(req *http.Request, service, reason string) *http.Response {
    body := awsBlockedBody{Error: "KM_AWS_BLOCKED", Service: service, Reason: reason}
    encoded, _ := json.Marshal(body)
    return goproxy.NewResponse(req, "application/json", http.StatusForbidden, string(encoded))
}

// callerCache memoizes uid → username lookups (best-effort getpwuid).
var callerCache sync.Map

func resolveCaller(uid uint32) string {
    if v, ok := callerCache.Load(uid); ok {
        return v.(string)
    }
    name := ""
    if u, err := user.LookupId(strconv.Itoa(int(uid))); err == nil && u != nil {
        name = u.Username
    }
    callerCache.Store(uid, name)
    return name
}

// platformUIDUnknown is the sentinel returned by GetCallerUID when the eBPF
// lookup fails. Treated as non-platform (gate applies) per 69-RESEARCH.md.
const platformUIDUnknown uint32 = 0xFFFFFFFF

// handleAWSRequest is the OnRequest handler body. Returns (req, nil) when the
// upstream call should proceed; (req, blockedResponse) when the proxy must
// short-circuit with a 403.
//
// callerUID: the UID owning the connection, looked up via GetCallerUID in
// transparent.go (Plan 69-06 wiring). Pass platformUIDUnknown when unavailable.
//
// platformUIDMax: KM_AWS_PLATFORM_UID_MAX boundary (default 1000).
func handleAWSRequest(req *http.Request, sandboxID string, opts *awsInspectionOptions, callerUID uint32, platformUIDMax uint32) (*http.Request, *http.Response) {
    service, region, signed := extractSigV4Service(req)
    if !signed {
        emitUnsigned(sandboxID, req)
        return req, nil
    }

    // Platform-uid carve-out: emit aws_api_platform and skip the gate.
    if callerUID != platformUIDUnknown && callerUID < platformUIDMax {
        emitPlatform(sandboxID, service, region, req, callerUID)
        return req, nil
    }

    matched := matchAllowlist(opts.allowlist, service)
    mode := opts.mode

    if matched {
        emitAllowed(sandboxID, service, region, req, mode)
        return req, nil
    }

    // Not matched. Behavior depends on mode + whether the allowlist was empty.
    reason := "not_in_allowlist"
    if len(opts.allowlist) == 0 {
        reason = "empty_allowlist"
    }
    emitBlocked(sandboxID, service, region, req, mode, reason)

    if mode == "enforce" {
        return req, AWSBlockedResponse(req, service, reason)
    }
    // observe mode: log but don't block
    return req, nil
}

func emitAllowed(sandboxID, service, region string, req *http.Request, mode string) {
    log.Info().
        Str("event_type", "aws_api_allowed").
        Str("sandbox_id", sandboxID).
        Str("service", service).
        Str("region", region).
        Str("host", req.Host).
        Str("method", req.Method).
        Str("path", req.URL.Path).
        Str("mode", mode).
        Msg("")
}

func emitBlocked(sandboxID, service, region string, req *http.Request, mode, reason string) {
    log.Info().
        Str("event_type", "aws_api_blocked").
        Str("sandbox_id", sandboxID).
        Str("service", service).
        Str("region", region).
        Str("host", req.Host).
        Str("method", req.Method).
        Str("path", req.URL.Path).
        Str("mode", mode).
        Str("reason", reason).
        Msg("")
}

func emitPlatform(sandboxID, service, region string, req *http.Request, uid uint32) {
    log.Info().
        Str("event_type", "aws_api_platform").
        Str("sandbox_id", sandboxID).
        Str("service", service).
        Str("region", region).
        Str("host", req.Host).
        Uint32("uid", uid).
        Str("caller", resolveCaller(uid)).
        Msg("")
}

func emitUnsigned(sandboxID string, req *http.Request) {
    log.Info().
        Str("event_type", "aws_api_unsigned").
        Str("sandbox_id", sandboxID).
        Str("host", req.Host).
        Str("method", req.Method).
        Str("path", req.URL.Path).
        Msg("")
}

// captureLogs replaces the global zerolog writer with a buffer and returns a
// restore func + decoder that yields each emitted event as a map.
func captureLogs(t *testing.T) (*bytes.Buffer, func()) {
    t.Helper()
    var buf bytes.Buffer
    origLogger := log.Logger
    log.Logger = zerolog.New(&buf).With().Timestamp().Logger()
    return &buf, func() { log.Logger = origLogger }
}

func decodeEvents(t *testing.T, buf *bytes.Buffer) []map[string]interface{} {
    t.Helper()
    var out []map[string]interface{}
    for _, line := range strings.Split(strings.TrimSpace(buf.String()), "\n") {
        if line == "" {
            continue
        }
        var m map[string]interface{}
        if err := json.Unmarshal([]byte(line), &m); err != nil {
            t.Fatalf("decode log line %q: %v", line, err)
        }
        out = append(out, m)
    }
    return out
}

func newSignedRequest(t *testing.T, host, authHeader string) *http.Request {
    t.Helper()
    u, _ := url.Parse("https://" + host + "/some/path")
    req := &http.Request{Method: "GET", URL: u, Header: http.Header{}, Host: host}
    if authHeader != "" {
        req.Header.Set("Authorization", authHeader)
    }
    return req
}

// parseSigV4Fixture is a tiny # label\n<value>\n parser for the testdata file.
func parseSigV4Fixture(t *testing.T, data []byte) map[string]string {
    t.Helper()
    out := map[string]string{}
    var label string
    for _, line := range strings.Split(string(data), "\n") {
        line = strings.TrimSpace(line)
        if line == "" {
            continue
        }
        if strings.HasPrefix(line, "#") {
            label = strings.TrimSpace(strings.TrimPrefix(line, "#"))
            continue
        }
        if label != "" {
            out[label] = line
            label = ""
        }
    }
    return out
}
