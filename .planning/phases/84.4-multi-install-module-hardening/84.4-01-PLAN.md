Type-assertion facts (HIGH confidence per RESEARCH.md):
- `*hclsyntax.LiteralValueExpr` — bare quoted string. `.Val` is cty.Value. Use `.AsString()`.
- `*hclsyntax.TemplateExpr` — `"${var.resource_prefix}-sandbox-containment"`. Has `Parts []Expression`.
- `*hclsyntax.ScopeTraversalExpr` — pure traversal `var.resource_prefix`. `.Traversal[0].Name == "var"`, `.Traversal[1].Name == "resource_prefix"`.

filepath.Glob CAN'T do `**` — use two-level glob `infra/modules/*/v*.0.0` and skip v1.0.0

    package terragrunt_test

    import (
        "fmt"
        "os"
        "path/filepath"
        "strings"
        "testing"

        "github.com/hashicorp/hcl/v2"
        "github.com/hashicorp/hcl/v2/hclsyntax"
    )

    // auditedAttrs: attribute names that MUST be prefix-templated, not literal.
    var auditedAttrs = map[string]bool{
        "name":           true,
        "creation_token": true,
        "alias":          true,
        "policy_name":    true,
    }

    func TestModuleNamesUseResourcePrefix(t *testing.T) {
        repoRoot := findRepoRoot(t)
        modules := walkModulesV2Plus(t, repoRoot)
        if len(modules) == 0 {
            t.Log("no v2.0.0+ modules found — Plan 02/03/04 will introduce them")
            return
        }
        allowlist := loadAllowlist(t, filepath.Join(repoRoot, "pkg/terragrunt/testdata/allowlist.txt"))
        for _, modDir := range modules {
            entries, err := os.ReadDir(modDir)
            if err != nil { t.Fatalf("ReadDir(%s): %v", modDir, err) }
            for _, e := range entries {
                if !strings.HasSuffix(e.Name(), ".tf") { continue }
                checkTFFile(t, filepath.Join(modDir, e.Name()), allowlist)
            }
        }
    }

    func walkModulesV2Plus(t *testing.T, repoRoot string) []string {
        t.Helper()
        base := filepath.Join(repoRoot, "infra", "modules")
        modules, err := os.ReadDir(base)
        if err != nil { t.Fatalf("ReadDir(%s): %v", base, err) }
        var result []string
        for _, m := range modules {
            if !m.IsDir() { continue }
            verDir := filepath.Join(base, m.Name())
            versions, _ := os.ReadDir(verDir)
            for _, v := range versions {
                if !v.IsDir() || !strings.HasPrefix(v.Name(), "v") { continue }
                if v.Name() == "v1.0.0" { continue }  // historical
                result = append(result, filepath.Join(verDir, v.Name()))
            }
        }
        return result
    }

    func checkTFFile(t *testing.T, path string, allowlist []string) {
        t.Helper()
        src, err := os.ReadFile(path)
        if err != nil { t.Fatalf("ReadFile(%s): %v", path, err) }
        file, diags := hclsyntax.ParseConfig(src, path, hcl.InitialPos)
        if diags.HasErrors() { t.Fatalf("parse %s: %v", path, diags) }
        body, ok := file.Body.(*hclsyntax.Body)
        if !ok { t.Fatalf("%s: body is not hclsyntax.Body", path) }
        for _, block := range body.Blocks {
            if block.Type != "resource" { continue }
            for attrName, attr := range block.Body.Attributes {
                if !auditedAttrs[attrName] { continue }
                checkAttrExpr(t, path, attrName, attr.Expr, attr.SrcRange, allowlist)
            }
        }
    }

    func checkAttrExpr(t *testing.T, path, attrName string, expr hclsyntax.Expression, srcRange hcl.Range, allowlist []string) {
        t.Helper()
        switch e := expr.(type) {
        case *hclsyntax.LiteralValueExpr:
            v := e.Val.AsString()
            if strings.Contains(v, "km-") && !isAllowlisted(v, allowlist) {
                t.Errorf("%s:%d: resource attribute %q = %q has hardcoded \"km-\" literal — use ${var.resource_prefix} instead",
                    path, srcRange.Start.Line, attrName, v)
            }
        case *hclsyntax.TemplateExpr:
            if !templateRefsResourcePrefix(e) {
                // FAIL only if any embedded literal part contains "km-"
                for _, part := range e.Parts {
                    if lit, ok := part.(*hclsyntax.LiteralValueExpr); ok {
                        v := lit.Val.AsString()
                        if strings.Contains(v, "km-") && !isAllowlisted(v, allowlist) {
                            t.Errorf("%s:%d: resource attribute %q template contains hardcoded \"km-\" literal %q — template must reference var.resource_prefix",
                                path, srcRange.Start.Line, attrName, v)
                        }
                    }
                }
            }
        }
    }

    // templateRefsResourcePrefix returns true if the template expression contains
    // any ScopeTraversalExpr that references var.resource_prefix.
    func templateRefsResourcePrefix(e *hclsyntax.TemplateExpr) bool {
        for _, part := range e.Parts {
            if st, ok := part.(*hclsyntax.ScopeTraversalExpr); ok {
                trav := st.Traversal
                if len(trav) >= 2 {
                    root, rOK := trav[0].(hcl.TraverseRoot)
                    attr, aOK := trav[1].(hcl.TraverseAttr)
                    if rOK && aOK && root.Name == "var" && attr.Name == "resource_prefix" {
                        return true
                    }
                }
            }
        }
        return false
    }

    // isAllowlisted checks whether a literal value matches an allow-list entry.
    // Conservative: substring match on a normalized form.
    func isAllowlisted(value string, allowlist []string) bool {
        for _, pattern := range allowlist {
            if pattern == "" || strings.HasPrefix(pattern, "#") { continue }
            // pattern is e.g. `km:manager = "km-cluster"` — extract the quoted RHS
            // and compare to value
            if rhsContainsValue(pattern, value) { return true }
        }
        return false
    }

    func rhsContainsValue(pattern, value string) bool {
        i := strings.Index(pattern, `"`)
        j := strings.LastIndex(pattern, `"`)
        if i < 0 || j <= i { return false }
        return pattern[i+1:j] == value
    }

    func loadAllowlist(t *testing.T, path string) []string {
        t.Helper()
        b, err := os.ReadFile(path)
        if err != nil { return nil } // empty allowlist is fine
        var out []string
        for _, line := range strings.Split(string(b), "\n") {
            line = strings.TrimSpace(line)
            if line == "" || strings.HasPrefix(line, "#") { continue }
            out = append(out, line)
        }
        return out
    }

    // findRepoRoot walks up from CWD looking for CLAUDE.md (the canonical anchor — see CLAUDE.md L3).
    func findRepoRoot(t *testing.T) string {
        t.Helper()
        cwd, err := os.Getwd()
        if err != nil { t.Fatalf("Getwd: %v", err) }
        dir := cwd
        for {
            if _, err := os.Stat(filepath.Join(dir, "CLAUDE.md")); err == nil {
                return dir
            }
            parent := filepath.Dir(dir)
            if parent == dir { t.Fatalf("CLAUDE.md not found from %s", cwd) }
            dir = parent
        }
    }
