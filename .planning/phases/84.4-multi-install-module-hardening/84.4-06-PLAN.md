<objective>
Extend `runBootstrapSharedSES` in `internal/app/cmd/bootstrap.go` with auto-detect +
auto-import for the three Route53 record types that the foundation SES module manages:
DKIM CNAMEs (count=3), MX (count=1), and `_amazonses` verification TXT (count=1).

The exact failure mode this closes: on a second install in a shared-domain account
(the `whereiskurt` probe was exactly this), the domain identity already exists in AWS
SES from the first install's `km bootstrap --shared-ses`. The DKIM tokens are stable
per domain — they're the SAME values across installs — but the second install's
foundation tfstate has no Route53 record resources yet. Without auto-import, the
second install's apply fails with "resource already exists" errors. Today operators
must run 3+ manual `terragrunt import` commands (documented in OPERATOR-GUIDE.md §
Phase 84.1). Phase 84.4 automates this.

This depends on Plan 00's `Runner.Import` method but does NOT depend on Plans
02/03/04/05 (it's a bootstrap.go change orthogonal to the v2.0.0 module work). Wave 2
parallelism: Plan 05 (live wiring flip) and Plan 06 (this plan) can both proceed once
Plan 00 is in.

Purpose: BOOTSTRAP-AUTO-IMPORT-EXISTING-DNS. Eliminates the manual `terragrunt import`
runbook for second installs in shared-domain accounts.

Output: bootstrap.go gains the import flow + unit tests. Live verification happens
during the fresh-prefix UAT (Plan 08) — the second install's `km bootstrap --shared-ses
--dry-run=false` should run end-to-end without manual operator action.
</objective>

    Step 4 — Implement the helper body in `internal/app/cmd/bootstrap.go`, replacing the

    ```go
    // autoImportFoundationSESRecords imports any pre-existing Route53 DKIM/MX/TXT
    // records that exist in AWS but are missing from foundation tfstate.
    // Idempotent: skips records already owned by state.
    // Returns nil if there's nothing to import (state is clean OR records don't exist in AWS).
    func autoImportFoundationSESRecords(
        ctx context.Context,
        runner importer,
        sesDir string,
        stateReader stateOwner,
        sesClient sesDkimGetter,
        r53Client route53Lister,
        emailDomain string,
        hostedZoneID string,
    ) error {
        // 1. Fetch DKIM tokens for the domain
        dkimOut, err := sesClient.GetIdentityDkimAttributes(ctx, &ses.GetIdentityDkimAttributesInput{
            Identities: []string{emailDomain},
        })
        if err != nil {
            return fmt.Errorf("get DKIM attributes for %s: %w", emailDomain, err)
        }
        attrs, ok := dkimOut.DkimAttributes[emailDomain]
        if !ok || len(attrs.DkimTokens) == 0 {
            fmt.Fprintf(os.Stderr, "warning: domain %s has no DKIM tokens yet — apply will create them\n", emailDomain)
            return nil
        }

        // 2. Import each DKIM CNAME if not already in state
        for i, token := range attrs.DkimTokens {
            if i >= 3 { break }  // module declares count = 3
            addr := fmt.Sprintf("aws_route53_record.dkim[%d]", i)
            owned, err := stateReader.StateOwns(addr)
            if err != nil { return fmt.Errorf("state check %s: %w", addr, err) }
            if owned {
                fmt.Fprintf(os.Stderr, "✓ DKIM[%d] already in state, skipping import\n", i)
                continue
            }
            id := fmt.Sprintf("%s_%s._domainkey.%s_CNAME", hostedZoneID, token, emailDomain)
            fmt.Fprintf(os.Stderr, "→ importing %s id=%s\n", addr, id)
            if err := runner.Import(ctx, sesDir, addr, id); err != nil {
                return fmt.Errorf("import %s: %w", addr, err)
            }
        }

        // 3. Conditionally import MX and TXT (look up in Route53 first)
        records, err := r53Client.ListResourceRecordSets(ctx, &route53.ListResourceRecordSetsInput{
            HostedZoneId:    aws.String(hostedZoneID),
            StartRecordName: aws.String(emailDomain),
        })
        if err != nil {
            return fmt.Errorf("list Route53 records for zone %s: %w", hostedZoneID, err)
        }

        var foundMX, foundTXT bool
        for _, rec := range records.ResourceRecordSets {
            name := aws.ToString(rec.Name)
            name = strings.TrimSuffix(name, ".")
            if rec.Type == r53types.RRTypeMx && name == emailDomain {
                foundMX = true
            }
            if rec.Type == r53types.RRTypeTxt && name == "_amazonses."+emailDomain {
                foundTXT = true
            }
        }

        if foundMX {
            addr := "aws_route53_record.mx[0]"
            owned, _ := stateReader.StateOwns(addr)
            if !owned {
                id := fmt.Sprintf("%s_%s_MX", hostedZoneID, emailDomain)
                fmt.Fprintf(os.Stderr, "→ importing %s id=%s\n", addr, id)
                if err := runner.Import(ctx, sesDir, addr, id); err != nil {
                    return fmt.Errorf("import MX: %w", err)
                }
            }
        }

        if foundTXT {
            addr := "aws_route53_record.ses_verification[0]"
            owned, _ := stateReader.StateOwns(addr)
            if !owned {
                // Note: DOUBLE underscore — record name starts with _amazonses
                id := fmt.Sprintf("%s__amazonses.%s_TXT", hostedZoneID, emailDomain)
                fmt.Fprintf(os.Stderr, "→ importing %s id=%s\n", addr, id)
                if err := runner.Import(ctx, sesDir, addr, id); err != nil {
                    return fmt.Errorf("import TXT: %w", err)
                }
            }
        }

        return nil
    }
    ```
