GitHub REST endpoints this plan wraps (all repo-scoped):
- POST   /repos/{owner}/{repo}/actions/runners/registration-token   — 201 returns {token, expires_at}
- POST   /repos/{owner}/{repo}/actions/runners/remove-token         — 201 returns {token, expires_at}
- DELETE /repos/{owner}/{repo}/actions/runners/{runner_id}          — 204 no content
- GET    /repos/{owner}/{repo}/actions/runners                      — 200 returns {runners: [{id, name, status, labels: [{name}]}]}

All four require `Authorization: Bearer <installation-token>` (NOT the App JWT — the
installation token is what `pkg/github/token.go:ExchangeForInstallationToken` produces).
For Phase 81 these calls run inside the token Lambda (Plan 81-02), which has the App
private key in SSM and will mint installation tokens on demand.

       ```go
       // validateRunner enforces Phase 81 RunnerSpec semantics: subset rule, maxRepos cap,
       // learnMode-incompatibility, allowedRepos-non-empty. Appends errors to errs.
       func validateRunner(p *SandboxProfile, errs []ValidationError) []ValidationError {
           if p.Spec.SourceAccess.GitHub == nil || p.Spec.SourceAccess.GitHub.Runner == nil {
               return errs
           }
           r := p.Spec.SourceAccess.GitHub.Runner
           if !r.Enabled {
               return errs
           }
           // Rule: allowedRepos non-empty
           if len(p.Spec.SourceAccess.GitHub.AllowedRepos) == 0 {
               errs = append(errs, ValidationError{
                   Path: "spec.sourceAccess.github.runner.enabled",
                   Message: "runner.enabled:true requires non-empty spec.sourceAccess.github.allowedRepos",
               })
           }
           // Rule: incompatible with learnMode
           if p.Spec.Observability.LearnMode {
               errs = append(errs, ValidationError{
                   Path: "spec.sourceAccess.github.runner.enabled",
                   Message: "runner.enabled:true is incompatible with observability.learnMode:true",
               })
           }
           // Rule: maxRepos cap (default 5)
           cap := r.MaxRepos
           if cap == 0 {
               cap = 5
           }
           if len(r.Repos) > cap {
               errs = append(errs, ValidationError{
                   Path: "spec.sourceAccess.github.runner.repos",
                   Message: fmt.Sprintf("runner.repos length %d exceeds maxRepos cap %d", len(r.Repos), cap),
               })
           }
           // Rule: subset of allowedRepos (with wildcard "org/*" expansion)
           for i, repo := range r.Repos {
               if !runnerRepoCoveredBy(repo, p.Spec.SourceAccess.GitHub.AllowedRepos) {
                   errs = append(errs, ValidationError{
                       Path: fmt.Sprintf("spec.sourceAccess.github.runner.repos[%d]", i),
                       Message: fmt.Sprintf("repo %q is not covered by spec.sourceAccess.github.allowedRepos", repo),
                   })
               }
           }
           return errs
       }

       // runnerRepoCoveredBy returns true if needle ("org/name") is covered by any
       // entry in haystack — exact match OR wildcard "org/*" prefix match.
       func runnerRepoCoveredBy(needle string, haystack []string) bool {
           for _, allow := range haystack {
               if allow == needle {
                   return true
               }
               if strings.HasSuffix(allow, "/*") {
                   prefix := strings.TrimSuffix(allow, "/*") + "/"
                   if strings.HasPrefix(needle, prefix) {
                       return true
                   }
               }
           }
           return false
       }
       ```

       ```go
       // NewRunnerClient constructs a RunnerClient with the given installation token and
       // HTTP transport. The httpClient parameter is the test/production seam:
       //   - Pass nil  → falls back to http.DefaultClient (production default).
       //   - Pass *http.Client → used verbatim (tests inject httptest server's client;

       func NewRunnerClient(token string, httpClient *http.Client) *RunnerClient {
           if httpClient == nil {
               httpClient = http.DefaultClient

               BaseURL: "https://api.github.com",
               HTTP:    httpClient,
               Token:   token,
       ```
