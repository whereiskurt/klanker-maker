<objective>
Fix three known km-recv parsing bugs and add automatic external-sender detection so sandboxes can read Gmail/Outlook email correctly:

1. **RFC 5322 folded header parsing.** `parse_headers()` currently uses `grep -i "^Header:"` which fails on multi-line folded headers (continuation lines starting with whitespace). Apply an `unfold_headers` awk pipeline to the raw email BEFORE `parse_headers` runs.

2. **multipart/alternative body extraction.** `extract_body()` only handles `multipart/mixed`; Gmail HTML email arrives as `multipart/alternative` (and may be nested inside `multipart/mixed` when there are attachments). Add `multipart/alternative` as a recognized top-level type, prefer `text/plain` over `text/html`, fall back to HTML-stripped output.

3. **`--from-external` automatic display hint.** When `X-KM-Sender-ID` is absent (external sender), show `[EXTERNAL]` in the table and add `"external": true` to the JSON output. No new CLI flag — detection is automatic (per research Pitfall 5).

This is the second of three sequential plans modifying `pkg/compiler/userdata.go`. Plan 01 (km-send `--no-sign`) is already complete; Plan 03 (km-mail-poller safe phrase) follows.

Purpose: Today, when a sandbox receives Gmail-originated email, km-recv either truncates the headers (folded subject), shows an empty body (multipart/alternative not recognized), or silently displays an unsigned message as if it were a sandbox peer. Phase 57 fixes all three.

    ```bash
    extract_body() {
      local raw="$1"
      local content_type="$HDR_CONTENT_TYPE"

      # Top-level dispatch
      case "$content_type" in
        *multipart/mixed*)
          extract_from_multipart "$raw" "$content_type"
          ;;
        *multipart/alternative*)
          extract_alternative "$raw" "$content_type"
          ;;
        *text/plain*)
          extract_after_blank_line "$raw"
          ;;
        *text/html*)
          extract_after_blank_line "$raw" | strip_html
          ;;
        *)
          extract_after_blank_line "$raw"
          ;;
      esac
    }

    # Extract body from multipart/mixed; if a part is itself multipart/alternative,
    # recurse into it. Otherwise prefer the first text/plain part.
    extract_from_multipart() {
      local raw="$1"
      local ct="$2"
      local boundary
      boundary=$(echo "$ct" | grep -oP 'boundary="?\K[^";]+' | head -1)
      [[ -z "$boundary" ]] && { extract_after_blank_line "$raw"; return; }

      # Iterate parts split by --$boundary; for each part, look at its Content-Type.
      # If part is multipart/alternative → recurse via extract_alternative on the part.
      # If part is text/plain → emit and return.
      # If part is text/html and no text/plain found → strip and emit.
      local parts
      parts=$(echo "$raw" | awk -v b="--$boundary" 'BEGIN{RS=""} {gsub("\r",""); print}')
      # Use python3 fallback if pure bash boundary-walk gets gnarly (research suggests sed/awk first)
      _walk_multipart_parts "$raw" "$boundary" "prefer_text_plain"
    }

    # Extract from multipart/alternative; prefer text/plain, fallback to text/html stripped.
    extract_alternative() {
      local raw="$1"
      local ct="$2"
      local boundary
      boundary=$(echo "$ct" | grep -oP 'boundary="?\K[^";]+' | head -1)
      [[ -z "$boundary" ]] && { extract_after_blank_line "$raw"; return; }
      _walk_multipart_parts "$raw" "$boundary" "prefer_text_plain"
    }

    # Helper: walk parts, prefer text/plain, fall back to text/html stripped.
    # If a nested part is multipart/alternative or multipart/related, recurse.
    _walk_multipart_parts() {
      local raw="$1"
      local boundary="$2"
      local mode="$3"
      local plain="" html="" nested_alt=""

      # Use awk to split on --boundary and capture each part's headers + body.
      local parts_dir
      parts_dir=$(mktemp -d)
      echo "$raw" | awk -v b="--$boundary" '
        BEGIN{ part=0; in_part=0 }
        index($0, b)==1 {
          part++; in_part=1
          out = "'"$parts_dir"'/part_" part
          next
        }
        in_part { print > out }
      '

      for f in "$parts_dir"/part_*; do
        [[ -f "$f" ]] || continue
        local part_ct
        part_ct=$(head -c 2048 "$f" | grep -i "^Content-Type:" | head -1 | tr -d '\r')
        local part_body
        part_body=$(awk 'BEGIN{h=1} h && /^$/ {h=0; next} !h {print}' "$f")
        case "$part_ct" in
          *multipart/alternative*|*multipart/related*)
            # Recurse: pass the part itself (which has its own Content-Type and boundary)
            local part_full
            part_full=$(cat "$f")
            local nested_b
            nested_b=$(echo "$part_ct" | grep -oP 'boundary="?\K[^";]+' | head -1)
            if [[ -n "$nested_b" ]]; then
              nested_alt=$(_walk_multipart_parts "$part_full" "$nested_b" "$mode")
              [[ -n "$nested_alt" ]] && plain="$nested_alt" && break
            fi
            ;;
          *text/plain*)
            plain="$part_body"
            break
            ;;
          *text/html*)
            [[ -z "$html" ]] && html="$part_body"
            ;;
        esac
      done
      rm -rf "$parts_dir"

      if [[ -n "$plain" ]]; then
        printf '%s' "$plain"
      elif [[ -n "$html" ]]; then
        printf '%s' "$html" | strip_html
      fi
    }

    # Basic HTML tag stripping (research Pattern 4 — sufficient for display).
    strip_html() {
      sed 's/<[^>]*>//g; s/&nbsp;/ /g; s/&amp;/\&/g; s/&lt;/</g; s/&gt;/>/g; s/&quot;/"/g'
    }

    # Helper for the simple text/plain top-level case.
    extract_after_blank_line() {
      awk 'BEGIN{h=1} h && /^[[:space:]]*$/ {h=0; next} !h {print}'
    }
    ```
