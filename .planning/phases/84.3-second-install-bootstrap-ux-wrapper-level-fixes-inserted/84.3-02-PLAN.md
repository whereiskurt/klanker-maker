// Variadic-last AWS interface for test injection — clone of bootstrap_kms_test.go KMSEnsureAPI pattern
type S3HeadBucketAPI interface {
    HeadBucket(ctx context.Context, params *s3.HeadBucketInput, optFns ...func(*s3.Options)) (*s3.HeadBucketOutput, error)
}

// Interactive HeadBucket probe with retry/edit/abort UX.
// - On 403: print suggestion `${initialName}-${accountID}`, prompt Y/edit/abort.
// - Y: HeadBucket the suggested name; on 403 → return error "pick a unique state_bucket via --state-bucket".
// - edit: prompt freeform; HeadBucket; fail-bail if 403.
// - abort: return error "aborted by operator".
// - On 404 or nil: return initialName, nil.
// stdin scanner: bufio.NewReader(os.Stdin) in real use; tests inject a *bufio.Reader over a string.
func probeStateBucketInteractive(
    ctx context.Context,
    initialName, accountID string,
    in *bufio.Reader, out io.Writer,
    s3client ...S3HeadBucketAPI,
) (string, error)

// Pure helper.
func deriveArtifactsBucket(prefix, accountID string) string {
    return fmt.Sprintf("%s-artifacts-%s", prefix, accountID)
}

// At-load placeholder rejection. Called from cfg-load path (NOT only configure).
// Rejects:
//   - strings.HasPrefix(name, "<") AND strings.HasSuffix(name, ">")  (angle-bracket)
//   - name == "km-artifacts-12345"  (literal example-yaml value)
// Returns nil for any other string (including empty — empty is handled by separate "required-field" check).
func validateArtifactsBucket(name string) error

// Pretty-prints the canonical bootstrap sequence.
// Used both as stdout block at end of runConfigure AND as `# ` header comments
// prepended to the generated km-config.yaml.
func nextStepsBlock() string

// Used during runConfigure: walk known KM_* env vars, compare to values being written
// to yaml, emit "exported in shell will shadow km-config.yaml in current session" WARN
// to stderr per drifted var. Does NOT block the wizard.
func warnShellEnvConflict(cfg *config.Config, w io.Writer)
