```go
// printTranscriptWarning prints a single stderr line when transcript streaming
// is enabled. Audience-containment warning for operator awareness; non-blocking.
// Phase 68.
func printTranscriptWarning(ctx context.Context, slackAPI SlackAPI, channelID string) {
    memberCount := "unknown"
    info, err := slackAPI.ChannelInfo(ctx, channelID)
    if err == nil && info != nil && info.NumMembers > 0 {
        memberCount = fmt.Sprintf("%d", info.NumMembers)
    }
    fmt.Fprintf(os.Stderr,
        "⚠ Slack transcript streaming enabled — full Claude transcripts (including tool I/O) will be posted to channel %s. Audience: %s Slack users.\n",
        channelID, memberCount,
    )
}
```
