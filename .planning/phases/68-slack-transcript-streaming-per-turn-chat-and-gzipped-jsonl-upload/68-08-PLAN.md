```go
    if env.Filename == "" || len(env.Filename) > 255 || strings.ContainsAny(env.Filename, "/\x00") {
```

```go
// S3GetterAdapter adapts an *s3.Client to the S3ObjectGetter interface.
// Phase 68 — bridge reads transcript objects from KM_ARTIFACTS_BUCKET.
type S3GetterAdapter struct {
    Client *s3.Client
    Bucket string
}

func (a *S3GetterAdapter) GetObject(ctx context.Context, key string) (io.ReadCloser, int64, error) {
    out, err := a.Client.GetObject(ctx, &s3.GetObjectInput{
        Bucket: aws.String(a.Bucket),
        Key:    aws.String(key),
    })
    if err != nil {
        return nil, 0, err
    }
    var sz int64
    if out.ContentLength != nil {
        sz = *out.ContentLength
    }
    return out.Body, sz, nil
}

// SlackFileUploaderAdapter adapts the slack.Client UploadFile (Plan 04) to the
// SlackFileUploader interface. Phase 68.
type SlackFileUploaderAdapter struct {
    Client *slack.Client
}

func (a *SlackFileUploaderAdapter) UploadFile(ctx context.Context, channel, threadTS, filename, contentType string, sizeBytes int64, body io.Reader) (string, string, error) {
    res, err := a.Client.UploadFile(ctx, channel, threadTS, filename, contentType, sizeBytes, body)
    if err != nil {
        return "", "", err
    }
    return res.FileID, res.Permalink, nil
}
```

```go
// Cold-start scope check: query auth.test once and capture X-OAuth-Scopes.
// Cached for the Lambda's lifetime (no per-request invocation).
func probeFilesWriteScope(ctx context.Context, baseURL, botToken string) (missing bool, err error) {
    req, _ := http.NewRequestWithContext(ctx, "POST", baseURL+"/api/auth.test", nil)
    req.Header.Set("Authorization", "Bearer "+botToken)
    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        return false, fmt.Errorf("auth.test: %w", err)  // err: caller decides — fail open (assume present) or fail closed
    }
    defer resp.Body.Close()
    scopes := resp.Header.Get("X-OAuth-Scopes")
    if scopes == "" {
        // Fall back to per-request behavior (no cached missing-flag); some Slack
        // installs return scopes only on /api.test or with non-standard headers.
        return false, nil
    }
    return !strings.Contains(scopes, "files:write"), nil
}
```
