```bash
if [[ "$event" == "PostToolUse" ]]; then
  # Read stdin payload (Claude provides session_id + transcript_path)
  payload=$(cat -)
  sid=$(echo "$payload" | jq -r '.session_id // ""')
  transcript_path=$(echo "$payload" | jq -r '.transcript_path // ""')
  if [[ -z "$sid" || -z "$transcript_path" || ! -f "$transcript_path" ]]; then
    echo "[km-notify-hook] PostToolUse: missing session_id/transcript_path" >&2
    exit 0
  fi

  # Resolve thread ts: env > /tmp cache > auto-create
  thread_cache="/tmp/km-slack-thread.${sid}"
  if [[ -n "${KM_SLACK_THREAD_TS:-}" ]]; then
    ts="$KM_SLACK_THREAD_TS"
  elif [[ -f "$thread_cache" ]]; then
    ts=$(cat "$thread_cache")
  else
    # Auto-thread-parent: post a parent message and cache its ts
    prompt_summary=$(echo "$payload" | jq -r '.tool_input.prompt // .tool_name // "auto-run"' | head -c 80)
    parent_body="/tmp/km-slack-parent-${sid}.txt"
    printf '🤖 [%s] turn started — %s\n' "${KM_SANDBOX_ID:-sandbox}" "$prompt_summary" > "$parent_body"
    parent_resp=$(/opt/km/bin/km-slack post --channel "$KM_SLACK_CHANNEL_ID" --body "$parent_body" 2>/dev/null || echo "")
    rm -f "$parent_body"
    ts=$(echo "$parent_resp" | grep -oP 'ts=\K[0-9.]+' | head -n1)
    if [[ -n "$ts" ]]; then
      echo "$ts" > "$thread_cache"
    else
      echo "[km-notify-hook] PostToolUse: failed to create thread parent; skipping" >&2
      exit 0
    fi
  fi

  # Offset tracking
  offset_file="/tmp/km-slack-stream.${sid}.offset"
  offset=0
  [[ -f "$offset_file" ]] && offset=$(cat "$offset_file")

  # Tail JSONL from offset
  new_lines=$(tail -c +$((offset + 1)) "$transcript_path" 2>/dev/null || echo "")
  new_offset=$(wc -c < "$transcript_path" 2>/dev/null | tr -d ' \t' || echo "$offset")

  if [[ -z "$new_lines" ]]; then
    echo "$new_offset" > "$offset_file"
    exit 0
  fi

  # Render new entries to a chat body
  body_file="/tmp/km-slack-stream-body-${sid}.txt"
  : > "$body_file"
  echo "$new_lines" | while IFS= read -r line; do
    [[ -z "$line" ]] && continue
    type=$(echo "$line" | jq -r '.type // ""')
    case "$type" in
      assistant)
        # Concatenate text blocks
        echo "$line" | jq -r '.message.content[]? | select(.type=="text") | .text' >> "$body_file"
        ;;
      tool_use)
        name=$(echo "$line" | jq -r '.name // ""')
        case "$name" in
          Edit)    one_liner=$(echo "$line" | jq -r '"\(.input.file_path // "?")"' | head -c 200) ;;
          Bash)    one_liner=$(echo "$line" | jq -r '.input.command // "?"' | head -c 200) ;;
          *)       one_liner=$(echo "$line" | jq -c '.input // {}' | head -c 80) ;;
        esac
        printf '🔧 %s: %s\n' "$name" "$one_liner" >> "$body_file"
        ;;
    esac
  done

  # If the body is empty after parsing, just advance the offset and exit
  if [[ ! -s "$body_file" ]]; then
    echo "$new_offset" > "$offset_file"
    rm -f "$body_file"
    exit 0
  fi

  # Post to Slack thread
  post_resp=$(/opt/km/bin/km-slack post --channel "$KM_SLACK_CHANNEL_ID" --thread "$ts" --body "$body_file" 2>&1 || echo "")
  rm -f "$body_file"
  posted_ts=$(echo "$post_resp" | grep -oP 'ts=\K[0-9.]+' | head -n1)

  if [[ -n "$posted_ts" && -n "${KM_SLACK_STREAM_TABLE:-}" ]]; then
    /opt/km/bin/km-slack record-mapping \
      --channel "$KM_SLACK_CHANNEL_ID" \
      --slack-ts "$posted_ts" \
      --offset "$offset" \
      --session "$sid" \
      2>/dev/null || echo "[km-notify-hook] record-mapping failed (non-fatal)" >&2
  fi

  # Advance offset
  echo "$new_offset" > "$offset_file"
  exit 0
fi
```

    var stdoutBuf, stderrBuf strings.Builder
    cmd.Stdout = &stdoutBuf
    cmd.Stderr = &stderrBuf
    err := cmd.Run()
    exitCode = 0
    if exitErr, ok := err.(*exec.ExitError); ok {
        exitCode = exitErr.ExitCode()
    }
    return stdoutBuf.String(), stderrBuf.String(), exitCode
}
