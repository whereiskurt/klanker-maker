package cmd

import (

// checkSlackTranscriptTableExists verifies the Phase 68 stream-messages
// DynamoDB table is provisioned. Returns WARN if missing/inactive; OK if ACTIVE.
func checkSlackTranscriptTableExists(ctx context.Context, doctor *Doctor) CheckResult {
    name := "slack_transcript_table_exists"
    tableName := doctor.Config.GetSlackStreamMessagesTableName()
    out, err := doctor.DDB.DescribeTable(ctx, &dynamodb.DescribeTableInput{
        TableName: aws.String(tableName),
    })
    if err != nil {
        return CheckResult{
            Name: name, Status: StatusWarn,
            Message: fmt.Sprintf("DDB DescribeTable %s: %v (run km init to provision)", tableName, err),
        }
    }
    if out.Table == nil || out.Table.TableStatus != ddbtypes.TableStatusActive {
        return CheckResult{
            Name: name, Status: StatusWarn,
            Message: fmt.Sprintf("table %s status=%s (expected ACTIVE)", tableName, out.Table.TableStatus),
        }
    }
    return CheckResult{
        Name: name, Status: StatusOK,
        Message: fmt.Sprintf("table %s ACTIVE", tableName),
    }
}

// checkSlackFilesWriteScope probes Slack auth.test for X-OAuth-Scopes; OK if
// files:write granted, WARN otherwise (transcript upload would 400 from bridge).
func checkSlackFilesWriteScope(ctx context.Context, doctor *Doctor) CheckResult {
    name := "slack_files_write_scope"
    botToken := doctor.SlackBotToken  // existing helper that reads /km/slack/bot-token from SSM
    if botToken == "" {
        return CheckResult{
            Name: name, Status: StatusWarn,
            Message: "no Slack bot token configured (skip — Phase 63 may not be set up)",
        }
    }
    req, _ := http.NewRequestWithContext(ctx, "POST", "https://slack.com/api/auth.test", nil)
    req.Header.Set("Authorization", "Bearer "+botToken)
    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        return CheckResult{
            Name: name, Status: StatusWarn,
            Message: fmt.Sprintf("auth.test HTTP: %v", err),
        }
    }
    defer resp.Body.Close()
    scopes := resp.Header.Get("X-OAuth-Scopes")
    if !strings.Contains(scopes, "files:write") {
        return CheckResult{
            Name: name, Status: StatusWarn,
            Message: "bot lacks files:write — re-auth Slack App at workspace admin",
        }
    }
    return CheckResult{
        Name: name, Status: StatusOK,
        Message: "bot has files:write scope",
    }
}

// checkSlackTranscriptStaleObjects lists S3 transcripts/ prefix and reports
// any prefix whose sandbox row no longer exists in km-sandboxes (cleanup advisory).
func checkSlackTranscriptStaleObjects(ctx context.Context, doctor *Doctor) CheckResult {
    name := "slack_transcript_stale_objects"
    bucket := doctor.Config.ArtifactsBucket
    if bucket == "" {
        return CheckResult{
            Name: name, Status: StatusWarn,
            Message: "no artifacts bucket configured (skip)",
        }
    }
    // List unique sandbox-id prefixes under transcripts/
    var prefixes []string
    var continuationToken *string
    for {
        out, err := doctor.S3.ListObjectsV2(ctx, &s3.ListObjectsV2Input{
            Bucket:            aws.String(bucket),
            Prefix:            aws.String("transcripts/"),
            Delimiter:         aws.String("/"),
            ContinuationToken: continuationToken,
        })
        if err != nil {
            return CheckResult{
                Name: name, Status: StatusWarn,
                Message: fmt.Sprintf("S3 ListObjectsV2: %v", err),
            }
        }
        for _, cp := range out.CommonPrefixes {
            if cp.Prefix != nil {
                prefixes = append(prefixes, *cp.Prefix) // e.g. "transcripts/sb-abc/"
            }
        }
        if out.IsTruncated == nil || !*out.IsTruncated {
            break
        }
        continuationToken = out.NextContinuationToken
    }

    // Get all live sandbox IDs from km-sandboxes (existing helper)
    liveIDs, err := doctor.ListSandboxIDs(ctx)
    if err != nil {
        return CheckResult{
            Name: name, Status: StatusWarn,
            Message: fmt.Sprintf("DDB Scan km-sandboxes: %v", err),
        }
    }
    liveSet := map[string]struct{}{}
    for _, id := range liveIDs {
        liveSet[id] = struct{}{}
    }

    var stale []string
    for _, p := range prefixes {
        // p is "transcripts/sb-abc/", extract the sandbox ID
        trimmed := strings.TrimPrefix(p, "transcripts/")
        sid := strings.TrimSuffix(trimmed, "/")
        if _, alive := liveSet[sid]; !alive {
            stale = append(stale, p)
        }
    }
    if len(stale) > 0 {
        return CheckResult{
            Name: name, Status: StatusWarn,
            Message: fmt.Sprintf("%d stale transcript prefixes: %s", len(stale), strings.Join(stale, ", ")),
        }
    }
    return CheckResult{
        Name: name, Status: StatusOK,
        Message: fmt.Sprintf("%d transcript prefixes; none stale", len(prefixes)),
    }
}
