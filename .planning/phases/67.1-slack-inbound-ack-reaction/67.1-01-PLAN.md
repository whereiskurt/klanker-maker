<objective>
Phase 67.1 closes the UAT gap where Slack users get no visual feedback that their inbound message was received until the agent boots and posts its first reply (10-60s for paused sandboxes). The fix: when the bridge Lambda accepts a message and successfully enqueues it to SQS, it adds a 👀 emoji reaction to the originating message within ~1 second via Slack `reactions.add` Web API.

This plan implements the **bridge-side code** for the ACK reaction feature:
1. New `Reactor` interface in `events_interfaces.go`
2. New `SlackReactorAdapter` in `aws_adapters.go` (mirrors SlackPosterAdapter, treats `already_reacted` as idempotent success)
3. New `Reactor` + `AckEmoji` fields on `EventsHandler` struct
4. New fire-and-forget goroutine block in `Handle()` immediately after the PauseHinter block — uses `msg.TS` (NOT `threadTS`)
5. Cold-start wiring in `cmd/km-slack-bridge/main.go` reading `KM_SLACK_ACK_EMOJI` (default `eyes`)
6. Wave-0 test scaffold: `fakeReactor` struct + extended `newHandler()` signature (~15 call sites need updating in events_handler_test.go)
7. New unit tests covering: happy path, reactor-failure-noop, bot-loop-skips, nil-reactor-back-compat, already_reacted idempotency

```go
// SlackReactorAdapter implements Reactor via direct HTTP to Slack's reactions.add Web API.
// Mirrors SlackPosterAdapter.call() shape for consistency. Treats already_reacted
// as idempotent success because Slack delivers events at-least-once.
type SlackReactorAdapter struct {
    HTTPClient *http.Client
    BaseURL    string          // defaults to "https://slack.com/api"; overridden in tests
    Tokens     BotTokenFetcher // SHARE THIS with SlackPosterAdapter to reuse the 15-min token cache
}

func (s *SlackReactorAdapter) getBaseURL() string {
    if s.BaseURL != "" {
        return s.BaseURL
    }
    return "https://slack.com/api"
}

// Add posts a reaction to a Slack message. Returns nil on success or
// already_reacted (idempotent). Returns ErrSlackRateLimited on HTTP 429.
// Returns a wrapped error for any other failure mode.
//
// emoji must be the bare emoji name without colons ("eyes", NOT ":eyes:").
// ts must be the originating message's TS field, NOT the thread root.
func (s *SlackReactorAdapter) Add(ctx context.Context, channel, ts, emoji string) error {
    token, err := s.Tokens.Fetch(ctx)
    if err != nil {
        return fmt.Errorf("bridge: get bot token for reactions.add: %w", err)
    }

    payload := map[string]any{
        "channel":   channel,
        "timestamp": ts,
        "name":      emoji,
    }
    b, err := json.Marshal(payload)
    if err != nil {
        return fmt.Errorf("bridge: marshal reactions.add: %w", err)
    }

    req, err := http.NewRequestWithContext(ctx, "POST", s.getBaseURL()+"/reactions.add", bytes.NewReader(b))
    if err != nil {
        return err
    }
    req.Header.Set("Authorization", "Bearer "+token)
    req.Header.Set("Content-Type", "application/json; charset=utf-8")

    hc := s.HTTPClient
    if hc == nil {
        hc = http.DefaultClient
    }

    resp, err := hc.Do(req)
    if err != nil {
        return err
    }
    defer resp.Body.Close()

    retryAfterHeader := resp.Header.Get("Retry-After")
    if rlErr := checkRateLimit(resp.StatusCode, retryAfterHeader, "reactions.add"); rlErr != nil {
        return rlErr
    }

    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return fmt.Errorf("bridge: read reactions.add response: %w", err)
    }
    var apiResp slackAPIResponse
    if err := json.Unmarshal(body, &apiResp); err != nil {
        return fmt.Errorf("bridge: decode reactions.add: %w", err)
    }
    if !apiResp.OK {
        if apiResp.Error == "already_reacted" {
            // Slack delivered the event twice; we already reacted on first delivery. Idempotent success.
            return nil
        }
        return fmt.Errorf("bridge: reactions.add: %s", apiResp.Error)
    }
    return nil
}
```

No new imports — `bytes`, `context`, `encoding/json`, `fmt`, `io`, `net/http` already present.

// TestEventsHandler_Reactor_HappyPath: valid message → Reactor.Add called once

func TestEventsHandler_Reactor_HappyPath(t *testing.T) {
    now := time.Now()
    h, sqs, _, _, _, _, reactor := newHandler(now)

    body := buildMessageEventBody(t, "C01234567", "1714280400.001", "", "U_HUMAN", "hello sandbox")

        },
    })

        t.Fatalf("expected 200, got %d (body=%s)", resp.StatusCode, resp.Body)
