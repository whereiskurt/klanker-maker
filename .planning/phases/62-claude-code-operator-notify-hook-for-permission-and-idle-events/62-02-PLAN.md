**Implementation note on the explicit-false problem.**
Go's `bool` zero value plus the `omitempty` yaml tag makes it impossible to distinguish "unset" from "explicit false" purely from the parsed struct. The CONTEXT.md spec wants both `true` and explicit `false` to emit the env var. Two options:

1. **Pragmatic (recommended for v1):** Treat unset and false identically. Always emit `KM_NOTIFY_ON_PERMISSION="0"`/`"1"` whenever `cli` is present. Skip the env var entirely only when `Spec.CLI` is nil. This deviates slightly from CONTEXT.md but matches user expectations: "if I configured cli, I configured these gates." Document the deviation in code comments and in the SUMMARY.

2. **Pointer-based (more faithful to spec):** Change `NotifyOnPermission` from `bool` to `*bool` in Plan 01. Allows nil = unset. Significantly more code churn (callers, tests, schema generation).

**Choose option 1.** Document in the Plan 02 SUMMARY and in 62-VERIFICATION.md as a tracked deviation. Test `TestUserDataNotifyEnvVars_ExplicitFalseStillEmitsZero` documents the resulting behavior. Test `TestUserDataNotifyEnvVars_NoneSet_NoEnvBlock` checks that a profile WITHOUT `cli` set emits no notify-env block.

# ============================================================
# Phase 62: Claude Code operator-notify hook (HOOK-01)
# Always installed; gated at runtime by KM_NOTIFY_ON_PERMISSION
# / KM_NOTIFY_ON_IDLE env vars (written conditionally below or
# injected by `km shell` / `km agent run` --notify-on-* flags).
# ============================================================
mkdir -p /opt/km/bin
cat > /opt/km/bin/km-notify-hook << 'KM_NOTIFY_HOOK_EOF'
#!/bin/bash
# km-notify-hook — emit signed email to operator on Claude Code hook events.
# Invoked by ~/.claude/settings.json hooks.Notification and hooks.Stop.
# First arg: event name (Notification or Stop). Stdin: Claude Code hook JSON.
# Always exits 0 — must never block Claude on email-send failures.
set -euo pipefail

event="${1:?event-name required}"

# 1. Gate check (env-var driven; defaults to off so unset = no-op).
case "$event" in
  Notification) [[ "${KM_NOTIFY_ON_PERMISSION:-0}" == "1" ]] || exit 0 ;;
  Stop)         [[ "${KM_NOTIFY_ON_IDLE:-0}"       == "1" ]] || exit 0 ;;
  *)            exit 0 ;;
esac

# 2. Cooldown check (per-sandbox, shared across both event types).
cooldown="${KM_NOTIFY_COOLDOWN_SECONDS:-0}"
last_file="/tmp/km-notify.last"
if [[ "$cooldown" -gt 0 && -f "$last_file" ]]; then
  last=$(cat "$last_file" 2>/dev/null || echo 0)
  now=$(date +%s)
  (( now - last < cooldown )) && exit 0
fi

# 3. Read hook payload from stdin (JSON).
payload=$(cat)

# 4. Build subject + body.
sandbox_id="${KM_SANDBOX_ID:-unknown}"
case "$event" in
  Notification)
    subject="[$sandbox_id] needs permission"
    msg=$(echo "$payload" | jq -r '.message // "(no message)"' 2>/dev/null || echo "(payload parse failed)")
    body_text="$msg"
    ;;
  Stop)
    subject="[$sandbox_id] idle"
    transcript=$(echo "$payload" | jq -r '.transcript_path // ""' 2>/dev/null || echo "")
    body_text=""
    if [[ -n "$transcript" && -f "$transcript" ]]; then
      body_text=$(tail -n 50 "$transcript" 2>/dev/null \
        | jq -r 'select(.type=="assistant") | .message.content[]? | select(.type=="text") | .text' 2>/dev/null \
        | tail -n 1)
    fi
    [[ -z "$body_text" ]] && body_text="(no recent assistant text)"
    ;;
esac

# 5. Build body file (km-send requires --body <file>, not stdin, per CLAUDE.md
#    — OpenSSL 3.5+ signing reliability).
body_file=$(mktemp /tmp/km-notify-body.XXXXXX)
{
  echo "$body_text"
  echo
  echo "---"
  echo "Attach:  km agent attach $sandbox_id"
  echo "Results: km agent results $sandbox_id"
} > "$body_file"

# 6. Send. Failure does not propagate.
to_args=()
[[ -n "${KM_NOTIFY_EMAIL:-}" ]] && to_args=(--to "$KM_NOTIFY_EMAIL")
if /opt/km/bin/km-send "${to_args[@]}" --subject "$subject" --body "$body_file"; then
  # 7. Update cooldown timestamp on success only.
  date +%s > "$last_file"
fi

rm -f "$body_file"
exit 0
KM_NOTIFY_HOOK_EOF
chmod +x /opt/km/bin/km-notify-hook
echo "[km-bootstrap] km-notify-hook installed at /opt/km/bin/km-notify-hook"

{{- if .NotifyEnv }}
# Phase 62: notify-hook env defaults from profile spec.cli (HOOK-03).
# Per-invocation overrides via `km shell --notify-*` / `km agent run --notify-*`
# inject env vars BEFORE the Claude process is launched, taking precedence.
cat > /etc/profile.d/km-notify-env.sh << 'KM_NOTIFY_ENV_EOF'
# Auto-generated by km compiler — Phase 62.
{{- range $key, $value := .NotifyEnv }}
export {{ $key }}="{{ $value }}"
{{- end }}
KM_NOTIFY_ENV_EOF

{{- end }}

```go
// Phase 62: mergeNotifyHookIntoSettings parses any user-supplied
// ~/.claude/settings.json from configFiles, appends the km-notify-hook
// commands to hooks.Notification and hooks.Stop arrays (preserving any
// existing entries), and writes the merged result back to configFiles.
// Returns an error iff the user-supplied JSON is malformed.
func mergeNotifyHookIntoSettings(configFiles map[string]string) (map[string]string, error) {
    const settingsPath = "/home/sandbox/.claude/settings.json"

    if configFiles == nil {
        configFiles = map[string]string{}
    }

    var settings map[string]interface{}
    if existing, ok := configFiles[settingsPath]; ok && strings.TrimSpace(existing) != "" {
        if err := json.Unmarshal([]byte(existing), &settings); err != nil {
            return nil, fmt.Errorf("invalid JSON in spec.execution.configFiles[%q]: %w", settingsPath, err)
        }
    }
    if settings == nil {
        settings = map[string]interface{}{}
    }

    hooks, _ := settings["hooks"].(map[string]interface{})
    if hooks == nil {
        hooks = map[string]interface{}{}
    }

    appendKMHook := func(eventName, cmd string) {
        kmEntry := map[string]interface{}{
            "hooks": []interface{}{
                map[string]interface{}{"type": "command", "command": cmd},
            },
        }
        existing, _ := hooks[eventName].([]interface{})
        hooks[eventName] = append(existing, kmEntry)
    }

    appendKMHook("Notification", "/opt/km/bin/km-notify-hook Notification")
    appendKMHook("Stop", "/opt/km/bin/km-notify-hook Stop")

    settings["hooks"] = hooks

    merged, err := json.MarshalIndent(settings, "", "  ")
    if err != nil {
        return nil, fmt.Errorf("marshal merged settings.json: %w", err)
    }
    configFiles[settingsPath] = string(merged)
    return configFiles, nil
}
```

#!/bin/bash
# Stub km-send for Phase 62 hook script tests.
# Activated via PATH override in `pkg/compiler/notify_hook_script_test.go`.
# Records the invocation and the body file contents to $KM_NOTIFY_TEST_LOG
# so the Go test can assert on subject, body, --to, etc.
#
# Usage (from Go test):
#   tmp := t.TempDir()
#   stub := filepath.Join(tmp, "km-send")
#   copy testdata/notify-hook-stub-km-send.sh -> stub
#   chmod +x stub
#   os.Setenv("PATH", tmp+":"+os.Getenv("PATH"))
#   os.Setenv("KM_NOTIFY_TEST_LOG", filepath.Join(tmp, "calls.log"))
#
# Exit code is controlled by KM_NOTIFY_TEST_FAIL: when set to "1", returns 1
# (used to verify hook always exits 0 even on send failure).

set -u
log="${KM_NOTIFY_TEST_LOG:-/dev/null}"

# Capture args verbatim, one per line, NUL-separated for safe parsing.
{
  printf '=== km-send call ===\n'
  printf 'args: %s\n' "$*"
  for arg in "$@"; do
    printf '  arg: %s\n' "$arg"
  done

  # Find --body <file> and dump its contents.
  body_file=""
  prev=""
  for arg in "$@"; do
    if [[ "$prev" == "--body" ]]; then
      body_file="$arg"
      break
    fi
    prev="$arg"
  done
  if [[ -n "$body_file" && -f "$body_file" ]]; then
    printf 'body_file: %s\n' "$body_file"
    printf 'body_contents_begin\n'
    cat "$body_file"
    printf '\nbody_contents_end\n'
  else
    printf 'body_file: (none or missing)\n'
  fi
  printf '=== end ===\n\n'
} >> "$log"

if [[ "${KM_NOTIFY_TEST_FAIL:-0}" == "1" ]]; then
  exit 1
fi
exit 0
