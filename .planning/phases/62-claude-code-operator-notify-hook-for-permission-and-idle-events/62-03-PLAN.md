```go
package compiler

import (
    "bytes"
    "io"
    "os"
    "os/exec"
    "path/filepath"
    "strings"
    "testing"
)

// extractNotifyHookScript renders user-data for a baseline profile, locates
// the km-notify-hook heredoc, and returns the script body as a string.
// Returns the body WITHOUT the surrounding heredoc delimiter lines.
func extractNotifyHookScript(t *testing.T) string {
    t.Helper()
    p := baseProfile()
    ud, err := generateUserData(p, /* same args as other userdata tests */)
    if err != nil { t.Fatalf("generateUserData: %v", err) }

    const open  = "cat > /opt/km/bin/km-notify-hook << 'KM_NOTIFY_HOOK_EOF'\n"
    const close = "\nKM_NOTIFY_HOOK_EOF\n"
    i := strings.Index(ud, open)
    if i < 0 { t.Fatalf("could not find hook script open delimiter in user-data") }
    j := strings.Index(ud[i+len(open):], close)
    if j < 0 { t.Fatalf("could not find hook script close delimiter in user-data") }
    return ud[i+len(open) : i+len(open)+j]
}

// setupHookEnv writes the hook script to a tmp file (chmod +x), copies the
// stub km-send into a tmp dir prepended to PATH, and returns:
//   - hookPath: the absolute path to the hook script
//   - logPath:  the absolute path the stub will append invocation records to
//   - cleanup:  a func to call via defer
//   - tmpdir:   the tmp dir (used by tests that need to manipulate /tmp/km-notify.last)
func setupHookEnv(t *testing.T) (hookPath, logPath, tmpdir string) {
    t.Helper()
    tmpdir = t.TempDir()

    // 1. Write hook script.
    body := extractNotifyHookScript(t)
    hookPath = filepath.Join(tmpdir, "km-notify-hook")
    if err := os.WriteFile(hookPath, []byte(body), 0o755); err != nil { t.Fatal(err) }

    // 2. Copy stub km-send into a "bin" dir we'll prepend to PATH.
    binDir := filepath.Join(tmpdir, "bin")
    if err := os.MkdirAll(binDir, 0o755); err != nil { t.Fatal(err) }
    stubSrc := filepath.Join("testdata", "notify-hook-stub-km-send.sh")
    stubBytes, err := os.ReadFile(stubSrc)
    if err != nil { t.Fatalf("read stub: %v", err) }
    stubDest := filepath.Join(binDir, "km-send")
    if err := os.WriteFile(stubDest, stubBytes, 0o755); err != nil { t.Fatal(err) }

    // The hook script calls /opt/km/bin/km-send by ABSOLUTE PATH. Tests must
    // either patch the script body or symlink /opt/km/bin/km-send. Since /opt
    // requires root, the test substitutes the script body to use plain `km-send`
    // (PATH-resolved). Do this in extractNotifyHookScript instead — replace
    // "/opt/km/bin/km-send" with "km-send" before writing.
    //   ^^^ Apply that substitution in extractNotifyHookScript() now.

    logPath = filepath.Join(tmpdir, "calls.log")

    // 3. Set PATH so stub is first.
    origPath := os.Getenv("PATH")
    t.Setenv("PATH", binDir+string(os.PathListSeparator)+origPath)
    t.Setenv("KM_NOTIFY_TEST_LOG", logPath)

    return hookPath, logPath, tmpdir
}

// runHook invokes the hook script with the given event arg, env vars (added
// on top of inherited), and stdin payload. Returns exit code, captured stdout,
// captured stderr.
func runHook(t *testing.T, hookPath, event string, env map[string]string, stdin string) (int, string, string) {
    t.Helper()
    cmd := exec.Command("bash", hookPath, event)
    cmd.Stdin = strings.NewReader(stdin)
    var out, errBuf bytes.Buffer
    cmd.Stdout = &out
    cmd.Stderr = &errBuf
    cmd.Env = os.Environ()
    for k, v := range env {
        cmd.Env = append(cmd.Env, k+"="+v)
    }
    err := cmd.Run()
    code := 0
    if exitErr, ok := err.(*exec.ExitError); ok {
        code = exitErr.ExitCode()
    } else if err != nil {
        t.Fatalf("runHook: %v\nstderr: %s", err, errBuf.String())
    }
    return code, out.String(), errBuf.String()
}

// loadFixture reads a testdata fixture and substitutes __TRANSCRIPT_PATH__.
func loadFixture(t *testing.T, name, transcriptPath string) string {
    t.Helper()
    data, err := os.ReadFile(filepath.Join("testdata", name))
    if err != nil { t.Fatal(err) }
    return strings.ReplaceAll(string(data), "__TRANSCRIPT_PATH__", transcriptPath)
}
```
