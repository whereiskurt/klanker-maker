```go
func checkPrefixCollision(ctx context.Context, p DoctorConfigProvider, lambdaCli LambdaGetFunctionClient) CheckResult {
    // 66-01 added GetResourcePrefix to the interface, so no type-assert needed.
    prefix := p.GetResourcePrefix()
    ttlName := prefix + "-ttl-handler"

    out, err := lambdaCli.GetFunction(ctx, &lambda.GetFunctionInput{FunctionName: aws.String(ttlName)})
    if err != nil {
        // ResourceNotFoundException = no collision (good).
        return CheckResult{Status: StatusOK, Message: "No prefix collision detected for " + ttlName}
    }
    _ = out
    // Function exists — without install-id tagging, we conservatively warn.
    // (Heuristic: the operator running this command IS the deployer, so this is informational.)
    return CheckResult{
        Status: StatusWarn,
        Message: fmt.Sprintf("Lambda %s already exists. Either you've already run km init (expected), OR another km install in this account is using the same resource_prefix. To run a second install, change resource_prefix in km-config.yaml.", ttlName),
    }
}

func checkEmailDomainMatchesSESIdentity(ctx context.Context, p DoctorConfigProvider, sesCli SESEmailIdentityClient) CheckResult {
    domain := p.GetEmailDomain()
    out, err := sesCli.GetEmailIdentity(ctx, &sesv2.GetEmailIdentityInput{EmailIdentity: aws.String(domain)})
    if err != nil {
        return CheckResult{
            Status: StatusWarn,
            Message: fmt.Sprintf("SES identity %q not found. If you changed email_subdomain after km init, verify the new domain in SES + add DKIM/MX records to Route53.", domain),
        }
    }
    if out.VerifiedForSendingStatus == nil || !*out.VerifiedForSendingStatus {
        return CheckResult{
            Status: StatusWarn,
            Message: fmt.Sprintf("SES identity %q exists but is not verified for sending.", domain),
        }
    }
    return CheckResult{Status: StatusOK, Message: "SES identity " + domain + " is verified"}
}
```
