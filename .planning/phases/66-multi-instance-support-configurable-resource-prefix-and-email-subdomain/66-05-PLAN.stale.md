km doctor check pattern (existing in doctor.go — extend the registered-checks slice):

```go
func checkPrefixCollision(ctx context.Context, p DoctorConfigProvider) doctorResult {
    // Detection: list Lambda functions in the configured region; if {prefix}-ttl-handler exists AND wasn't deployed by this install, warn.
    // Heuristic: read tags on the Lambda; if km:install-id tag mismatches our install ID (or is absent), it's another install.
    prefix := p.GetResourcePrefix()
    ttlName := prefix + "-ttl-handler"

    cli := lambda.NewFromConfig(awsCfg)
    out, err := cli.GetFunction(ctx, &lambda.GetFunctionInput{FunctionName: aws.String(ttlName)})
    if err != nil {
        // ResourceNotFoundException = no collision (good).
        return doctorOK("No prefix collision detected for " + ttlName)
    }
    // Function exists — check tags. If tagged with our install, it's us. If not, warn.
    // (Install-id tagging is out of scope; for now, warn whenever the Lambda exists and we're not the deployer.)
    return doctorWarn("Lambda " + ttlName + " already exists. Either you've already run km init, OR another km install in this account is using the same resource_prefix. To avoid collisions, change resource_prefix in km-config.yaml.")
}

func checkEmailDomainMatchesSESIdentity(ctx context.Context, p DoctorConfigProvider) doctorResult {
    domain := p.GetEmailDomain()
    cli := sesv2.NewFromConfig(awsCfg)
    out, err := cli.GetEmailIdentity(ctx, &sesv2.GetEmailIdentityInput{EmailIdentity: aws.String(domain)})
    if err != nil {
        return doctorWarn(fmt.Sprintf("SES identity %q not found. If you changed email_subdomain after km init, you need to verify the new domain in SES + add DKIM/MX records to Route53.", domain))
    }
    if out.VerifiedForSendingStatus == nil || !*out.VerifiedForSendingStatus {
        return doctorWarn(fmt.Sprintf("SES identity %q exists but is not verified for sending.", domain))
    }
    return doctorOK("SES identity " + domain + " is verified")
}
```
