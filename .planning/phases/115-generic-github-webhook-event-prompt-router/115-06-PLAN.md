       if [ "$KIND" = "issue_comment" ] || [ -z "$KIND" ]; then
           PREAMBLE="[GitHub Comment Trigger]
       Repository: $REPO
       PR: #$NUMBER
       ... (existing body, including the git fetch origin pull/${NUMBER}/head block at 2256) ..."
       else
           PREAMBLE="[GitHub Event Trigger]
       Repository: $REPO
       Event: $KIND / $ACTION
       Sender: $SENDER
       URL: $HTML_URL

       --- Task ---
       $COMMENT_BODY"
       fi

   + fetch to Phase 114. Empty KIND (old envelope) falls into the issue_comment branch (safe default).
