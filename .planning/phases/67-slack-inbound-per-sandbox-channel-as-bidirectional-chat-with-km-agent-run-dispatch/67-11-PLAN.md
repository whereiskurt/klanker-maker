```bash

    subject="[$sandbox_id] idle"

    body_text=""
    if [[ -n "$transcript" && -f "$transcript" ]]; then

# Phase 67 Gap A fix: post the canonical .result text to Slack from the
# poller — the Stop hook's transcript-JSONL scraping is unreliable in
# `claude -p` mode (transcript may be un-flushed or differently shaped).
# Read .result directly from output.json and post via km-slack.
# Set KM_SLACK_INBOUND_REPLY_HANDLED=1 so the Stop hook (which fires
# inside the same process tree via the Claude settings.json hooks
# config) skips its own Slack branch — exactly one post per turn.
if [ -n "${KM_SLACK_CHANNEL_ID:-}" ] && [ -n "${KM_SLACK_BRIDGE_URL:-}" ]; then
  POST_BODY="$RESULT_TEXT"
  [ -z "$POST_BODY" ] && POST_BODY="(no recent assistant text)"
  POST_FILE=$(mktemp /tmp/km-slack-inbound-post.XXXXXX)
  echo "$POST_BODY" > "$POST_FILE"
  export KM_SLACK_INBOUND_REPLY_HANDLED=1
  if /opt/km/bin/km-slack post \
       --channel "$KM_SLACK_CHANNEL_ID" \
       --subject "" \
       --thread "$THREAD_TS" \
       --body "$POST_FILE" 2>>"$RUN_DIR/stderr.log"; then
    echo "[km-slack-inbound-poller] Posted reply to Slack (thread=$THREAD_TS, len=${#RESULT_TEXT})"
  else
    echo "[km-slack-inbound-poller] WARN: km-slack post failed for thread=$THREAD_TS" >&2
  fi
  rm -f "$POST_FILE"
fi
```
