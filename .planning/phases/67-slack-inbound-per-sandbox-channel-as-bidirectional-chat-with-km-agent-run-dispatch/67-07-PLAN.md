<objective>
Round out the Slack-inbound lifecycle: post the operator-signed "ready" announcement after `km create` succeeds (anchoring the default-thread record in km-slack-threads), and implement the drain sequence for `km destroy` (stop poller → wait ≤30s for in-flight agent run → final post → delete queue → delete thread rows). The previous plan (67-06) left the SQS lifecycle hooked into create-only; this plan completes the round-trip.

Purpose: Without a "ready" announcement, the operator has nothing to reply to in a fresh sandbox channel — they'd have to start a top-level post, which works but is awkward UX. Without drain, `km destroy` race-deletes the queue while the poller is mid-turn, dropping the user's last message and leaving an unfinished tmux session.

Output: One new helper function in create_slack_inbound.go + one new file destroy_slack_inbound.go + edits to create.go and destroy.go + tests.
</objective>

```go
// postReadyAnnouncement posts the "Sandbox <id> ready" message via the
// existing Phase 63 operator-signed bridge `post` action and records the
// resulting message_ts in km-slack-threads with empty claude_session_id.
// First user reply directly to that message starts a fresh Claude session.
//
// Returns nil on best-effort failure (logs WARN) — failing to post the ready
// hint does NOT abort km create. The user can always start a top-level post.
func postReadyAnnouncement(ctx context.Context, deps slackInboundDeps, channelID string) error {
    if !deps.Profile.Spec.CLI.NotifySlackInboundEnabled {
        return nil
    }
    if channelID == "" {
        return fmt.Errorf("postReadyAnnouncement: empty channelID")
    }

    // Compose announcement. Wording per Claude's discretion (CONTEXT.md).
    msg := fmt.Sprintf(
        "Sandbox `%s` ready. Reply here or in any thread to give it a task.\n"+
            "_Profile: %s · Region: %s · Reply with a prompt to start a Claude turn._",
        deps.SandboxID, deps.Profile.Metadata.Name, deps.Cfg.Region,
    )

    // Post via existing operator-signed bridge action (same path used by
    // km slack test and km destroy final-post).
    messageTS, err := deps.PostOperatorSigned(ctx, channelID, msg)
    if err != nil {
        // Non-fatal: log WARN and continue. User can still send a top-level
        // post to start a conversation.
        fmt.Fprintf(os.Stderr, "⚠ Slack: ready announcement failed: %v (km create continues)\n", err)
        return nil
    }
    fmt.Fprintf(os.Stderr, "✓ Slack: posted ready announcement (ts=%s)\n", messageTS)

    // Anchor in km-slack-threads with empty claude_session_id. First reply
    // to this thread starts a fresh session.
    if err := deps.UpsertSlackThread(ctx, channelID, messageTS, deps.SandboxID); err != nil {
        fmt.Fprintf(os.Stderr, "⚠ Slack: thread anchor write failed: %v (non-fatal)\n", err)
    }
    return nil
}
```

```go
package cmd

import (
    "context"
    "fmt"
    "os"
    "time"

    awspkg "github.com/klankermaker/klankermaker/pkg/aws"
    awssdk "github.com/aws/aws-sdk-go-v2/aws"
    "github.com/aws/aws-sdk-go-v2/service/dynamodb"
    ddbtypes "github.com/aws/aws-sdk-go-v2/service/dynamodb/types"
)

// drainTimeout caps the wait for in-flight km agent run before km destroy
// proceeds. CONTEXT.md spec: best-effort, ~30s.
const drainTimeout = 30 * time.Second

// destroyInboundDeps bundles the pieces needed by drainSlackInbound.
type destroyInboundDeps struct {
    SandboxID    string
    InstanceID   string
    QueueURL     string  // empty if inbound was never enabled — drain becomes a no-op
    ChannelID    string
    BridgeURL    string

    SQS                   awspkg.SQSClient
    DDB                   awspkg.DDBQueryDeleteAPI // see below — small new interface in pkg/aws
    SlackThreadsTableName string

    // StopPoller stops the km-slack-inbound-poller systemd unit on the
    // sandbox via SSM SendCommand. Implementation lives in the existing
    // SSM helper module (see destroy_slack.go pattern).
    StopPoller func(ctx context.Context, instanceID string) error

    // WaitForAgentRunIdle returns nil when no /workspace/.km-agent/runs/<latest>
    // tmux session is active, error if drain timeout elapsed.
    WaitForAgentRunIdle func(ctx context.Context, instanceID string, deadline time.Time) error

    // PostFinalMessage is the existing destroy_slack.go function that posts
    // the "Sandbox X destroyed at <ts>" message via operator-signed bridge.
    PostFinalMessage func(ctx context.Context, channelID, body string) error
}

// drainSlackInbound is the orchestrator for km destroy's Slack-inbound path.
// Sequence (CONTEXT.md):
//  1. Stop systemd poller — prevent new turns from starting.
//  2. Wait up to 30s for any in-flight `km agent run` to finish.
//  3. (caller responsible for the existing Phase 63 final-post)
//  4. Delete SQS queue (drops any unprocessed messages — bounded by drain wait).
//  5. Delete km-slack-threads rows for this channel_id.
//
// Each step is best-effort: failures are logged but do NOT block km destroy.
func drainSlackInbound(ctx context.Context, deps destroyInboundDeps) {
    if deps.QueueURL == "" {
        // Sandbox doesn't have inbound — no-op.
        return
    }
    fmt.Fprintf(os.Stderr, "Slack inbound drain: starting (queue=%s)\n", deps.QueueURL)

    // Step 1: stop poller
    if deps.StopPoller != nil {
        stopCtx, cancel := context.WithTimeout(ctx, 10*time.Second)
        defer cancel()
        if err := deps.StopPoller(stopCtx, deps.InstanceID); err != nil {
            fmt.Fprintf(os.Stderr, "⚠ Slack inbound drain: stop poller failed: %v (continuing)\n", err)
        } else {
            fmt.Fprintf(os.Stderr, "✓ Slack inbound drain: poller stopped\n")
        }
    }

    // Step 2: wait for in-flight agent run
    if deps.WaitForAgentRunIdle != nil {
        deadline := time.Now().Add(drainTimeout)
        waitCtx, cancel := context.WithDeadline(ctx, deadline)
        defer cancel()
        if err := deps.WaitForAgentRunIdle(waitCtx, deps.InstanceID, deadline); err != nil {
            fmt.Fprintf(os.Stderr, "⚠ Slack inbound drain: agent-run still active after %s (continuing)\n", drainTimeout)
        } else {
            fmt.Fprintf(os.Stderr, "✓ Slack inbound drain: agent-run idle\n")
        }
    }

    // Step 4 (numbered per CONTEXT.md): delete SQS queue
    if err := awspkg.DeleteSlackInboundQueue(ctx, deps.SQS, deps.QueueURL); err != nil {
        fmt.Fprintf(os.Stderr, "⚠ Slack inbound drain: queue delete failed: %v (continuing)\n", err)
    } else {
        fmt.Fprintf(os.Stderr, "✓ Slack inbound drain: queue deleted\n")
    }

    // Step 5: delete km-slack-threads rows for this channel_id
    if deps.ChannelID != "" {
        if err := cleanupSlackThreads(ctx, deps); err != nil {
            fmt.Fprintf(os.Stderr, "⚠ Slack inbound drain: threads cleanup failed: %v\n", err)
        }
    }
}

// cleanupSlackThreads queries km-slack-threads for all rows with the given
// channel_id and deletes them in batches of 25 (DDB BatchWriteItem max).
func cleanupSlackThreads(ctx context.Context, deps destroyInboundDeps) error {
    // Query by partition key (channel_id is PK in our schema)
    out, err := deps.DDB.Query(ctx, &dynamodb.QueryInput{
        TableName:              awssdk.String(deps.SlackThreadsTableName),
        KeyConditionExpression: awssdk.String("channel_id = :cid"),
        ExpressionAttributeValues: map[string]ddbtypes.AttributeValue{
            ":cid": &ddbtypes.AttributeValueMemberS{Value: deps.ChannelID},
        },
    })
    if err != nil {
        return fmt.Errorf("query threads for channel %s: %w", deps.ChannelID, err)
    }
    deleted := 0
    for _, item := range out.Items {
        ts, _ := item["thread_ts"].(*ddbtypes.AttributeValueMemberS)
        if ts == nil { continue }
        _, derr := deps.DDB.DeleteItem(ctx, &dynamodb.DeleteItemInput{
            TableName: awssdk.String(deps.SlackThreadsTableName),
            Key: map[string]ddbtypes.AttributeValue{
                "channel_id": &ddbtypes.AttributeValueMemberS{Value: deps.ChannelID},
                "thread_ts":  ts,
            },
        })
        if derr != nil {
            fmt.Fprintf(os.Stderr, "⚠ Slack inbound drain: failed to delete thread %s/%s: %v\n", deps.ChannelID, ts.Value, derr)
            continue
        }
        deleted++
    }
    fmt.Fprintf(os.Stderr, "✓ Slack inbound drain: cleaned up %d thread row(s)\n", deleted)
    return nil
}
```
