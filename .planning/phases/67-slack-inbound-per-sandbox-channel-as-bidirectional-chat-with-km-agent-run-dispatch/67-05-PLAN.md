// SQSSendMessageAPI matches the subset of *sqs.Client used by SQSAdapter.
type SQSSendMessageAPI interface {
    SendMessage(ctx context.Context, in *sqs.SendMessageInput, opts ...func(*sqs.Options)) (*sqs.SendMessageOutput, error)
}

type SQSAdapter struct {
    Client SQSSendMessageAPI
}

func (a *SQSAdapter) Send(ctx context.Context, queueURL, body, groupID, dedupID string) error {
    if queueURL == "" {
        return fmt.Errorf("sqs send: empty queue url")
    }
    _, err := a.Client.SendMessage(ctx, &sqs.SendMessageInput{
        QueueUrl:               awssdk.String(queueURL),
        MessageBody:            awssdk.String(body),
        MessageGroupId:         awssdk.String(groupID),
        MessageDeduplicationId: awssdk.String(dedupID),
    })
    if err != nil {
        return fmt.Errorf("sqs send to %s: %w", queueURL, err)
    }
    return nil
}

// DDBQueryGetPutAPI matches the subset of *dynamodb.Client used by DDBThreadStore.
type DDBQueryGetPutAPI interface {
    GetItem(ctx context.Context, in *dynamodb.GetItemInput, opts ...func(*dynamodb.Options)) (*dynamodb.GetItemOutput, error)
    PutItem(ctx context.Context, in *dynamodb.PutItemInput, opts ...func(*dynamodb.Options)) (*dynamodb.PutItemOutput, error)
    Query(ctx context.Context, in *dynamodb.QueryInput, opts ...func(*dynamodb.Options)) (*dynamodb.QueryOutput, error)
}

type DDBThreadStore struct {
    Client    DDBQueryGetPutAPI
    TableName string // "km-slack-threads"
}

func (s *DDBThreadStore) Get(ctx context.Context, channelID, threadTS string) (string, error) {
    out, err := s.Client.GetItem(ctx, &dynamodb.GetItemInput{
        TableName: awssdk.String(s.TableName),
        Key: map[string]ddbtypes.AttributeValue{
            "channel_id": &ddbtypes.AttributeValueMemberS{Value: channelID},
            "thread_ts":  &ddbtypes.AttributeValueMemberS{Value: threadTS},
        },
    })
    if err != nil {
        return "", fmt.Errorf("threads get: %w", err)
    }
    if v, ok := out.Item["claude_session_id"]; ok {
        if s, ok := v.(*ddbtypes.AttributeValueMemberS); ok {
            return s.Value, nil
        }
    }
    return "", nil
}

func (s *DDBThreadStore) Upsert(ctx context.Context, channelID, threadTS, sandboxID string) error {
    now := time.Now()
    nowISO := now.UTC().Format(time.RFC3339)
    ttlExpiry := strconv.FormatInt(now.Add(30*24*time.Hour).Unix(), 10)
    _, err := s.Client.PutItem(ctx, &dynamodb.PutItemInput{
        TableName: awssdk.String(s.TableName),
        Item: map[string]ddbtypes.AttributeValue{
            "channel_id":  &ddbtypes.AttributeValueMemberS{Value: channelID},
            "thread_ts":   &ddbtypes.AttributeValueMemberS{Value: threadTS},
            "sandbox_id":  &ddbtypes.AttributeValueMemberS{Value: sandboxID},
            "created_at":  &ddbtypes.AttributeValueMemberS{Value: nowISO},
            "last_turn_ts": &ddbtypes.AttributeValueMemberS{Value: nowISO},
            "turn_count":  &ddbtypes.AttributeValueMemberN{Value: "0"},
            "ttl_expiry":  &ddbtypes.AttributeValueMemberN{Value: ttlExpiry},
        },
        ConditionExpression: awssdk.String("attribute_not_exists(channel_id)"),
    })
    if err != nil {
        // ConditionalCheckFailed = row already exists. That's the idempotent
        // happy path — bridge should NOT overwrite poller-written session_id.
        var ccfe *ddbtypes.ConditionalCheckFailedException
        if errors.As(err, &ccfe) {
            return nil
        }
        return fmt.Errorf("threads upsert: %w", err)
    }
    return nil
}

func (f *DDBSandboxByChannel) FetchByChannel(ctx context.Context, channelID string) (SandboxRoutingInfo, error) {
    out, err := f.Client.Query(ctx, &dynamodb.QueryInput{
        TableName:              awssdk.String(f.TableName),
        IndexName:              awssdk.String(f.IndexName),
        KeyConditionExpression: awssdk.String("slack_channel_id = :cid"),
        ExpressionAttributeValues: map[string]ddbtypes.AttributeValue{
            ":cid": &ddbtypes.AttributeValueMemberS{Value: channelID},
        },
        Limit: awssdk.Int32(1),
    })
    if err != nil {
        return SandboxRoutingInfo{}, fmt.Errorf("sandbox-by-channel query: %w", err)
    }
    if len(out.Items) == 0 {
        return SandboxRoutingInfo{}, nil // unknown channel — handler logs warn
    }
    item := out.Items[0]
    info := SandboxRoutingInfo{}
    if v, ok := item["sandbox_id"].(*ddbtypes.AttributeValueMemberS); ok {
        info.SandboxID = v.Value
    }
    if v, ok := item["slack_inbound_queue_url"].(*ddbtypes.AttributeValueMemberS); ok {
        info.QueueURL = v.Value
    }
    if v, ok := item["state"].(*ddbtypes.AttributeValueMemberS); ok {
        info.Paused = v.Value == "paused" || v.Value == "stopped"
    }
    return info, nil
}

// ==== SigningSecretFetcher (mirrors SSMBotTokenFetcher) ====

type SSMSigningSecretFetcher struct {
    Client   SSMGetParameter // existing interface in aws_adapters.go
    Path     string
    CacheTTL time.Duration

    mu       sync.Mutex
    value    string
    fetched  time.Time
}

func (f *SSMSigningSecretFetcher) Fetch(ctx context.Context) (string, error) {
    f.mu.Lock()
    defer f.mu.Unlock()
    if f.value != "" && time.Since(f.fetched) < f.CacheTTL {
        return f.value, nil
    }
    out, err := f.Client.GetParameter(ctx, &ssm.GetParameterInput{
        Name:           awssdk.String(f.Path),
        WithDecryption: awssdk.Bool(true),
    })
    if err != nil {
        return "", fmt.Errorf("get signing secret %s: %w", f.Path, err)
    }
    if out.Parameter == nil || out.Parameter.Value == nil {
        return "", fmt.Errorf("signing secret %s empty", f.Path)
    }
    f.value = *out.Parameter.Value
    f.fetched = time.Now()
    return f.value, nil
}

// ==== BotUserIDFetcher (calls auth.test once, caches for Lambda lifetime) ====

type SlackAuthTestAPI interface {
    AuthTest(ctx context.Context, token string) (userID string, err error)
}

type CachedBotUserIDFetcher struct {
    SlackAPI    SlackAuthTestAPI
    TokenFetcher SlackTokenFetcher

    mu     sync.Mutex
    value  string
    fetched time.Time
    ttl    time.Duration // e.g. 1h; bot_user_id changes only on token rotation
}

func (f *CachedBotUserIDFetcher) Fetch(ctx context.Context) (string, error) {
    f.mu.Lock()
    defer f.mu.Unlock()
    if f.value != "" && time.Since(f.fetched) < f.ttl {
        return f.value, nil
    }
    token, err := f.TokenFetcher.Fetch(ctx)
    if err != nil {
        return "", fmt.Errorf("bot_user_id: token fetch: %w", err)
    }
    uid, err := f.SlackAPI.AuthTest(ctx, token)
    if err != nil {
        return "", fmt.Errorf("bot_user_id: auth.test: %w", err)
    }
    f.value = uid
    f.fetched = time.Now()
    return f.value, nil
}

// ==== PauseHintPoster (DDBPauseHinter) — CONTEXT.md "Edge Cases" + checker BLOCKER ====

// DDBUpdateItemAPI is the subset of *dynamodb.Client used by DDBPauseHinter.
// (Note: DDBQueryGetPutAPI above does not include UpdateItem; this is intentional
// — the threads adapter doesn'''t do UpdateItem, only the pause-hinter does.)
type DDBUpdateItemAPI interface {
    DDBQueryGetPutAPI
    UpdateItem(ctx context.Context, in *dynamodb.UpdateItemInput, opts ...func(*dynamodb.Options)) (*dynamodb.UpdateItemOutput, error)
}

// PostHintFunc abstracts the bridge'''s operator-signed `post` action so the
// adapter can be unit-tested without round-tripping HTTP.  Plan 67-05 wires
// this in main.go to a closure that builds an operator-signed envelope and
// invokes the Phase 63 outbound path.
type PostHintFunc func(ctx context.Context, channelID, threadTS, text string) error

// DDBPauseHinter posts a one-time "paused; queued" hint into the user'''s
// Slack thread when the destination sandbox is paused, gated by a 1h
// cooldown stored as last_pause_hint_ts on the km-sandboxes row.
//
// Cooldown logic:
//   1. Read last_pause_hint_ts via GetItem on km-sandboxes/{sandbox_id}.
//   2. If absent OR (now - last_pause_hint_ts) > 3600s: post the hint AND
//      conditional-write last_pause_hint_ts=now (LWT) to absorb the
//      bridge cold-start race (two concurrent invocations both seeing an
//      absent attribute).  If the conditional write loses the race, that'''s
//      benign — a hint was already posted; we silently skip.
//   3. If cooldown active: return nil without posting.
//
// The adapter does NOT know about Slack channel/thread mapping — it'''s
// purely sandbox-keyed.  EventsHandler passes channel_id + thread_ts; the
// adapter resolves to sandbox_id via the same SandboxByChannelFetcher used
// in the main flow (injected as a dependency for testability).
type DDBPauseHinter struct {
    Client                DDBUpdateItemAPI
    SandboxesTableName    string                  // "km-sandboxes"
    SandboxByChannel      SandboxByChannelFetcher // resolves channel→sandbox_id
    Post                  PostHintFunc
    HintText              string                  // "Sandbox is paused; message queued. Run `km resume <id>` to wake it up."
    CooldownSeconds       int64                   // 3600 (1h per CONTEXT.md)
    Now                   func() time.Time        // injectable for tests; defaults to time.Now
}

func (a *DDBPauseHinter) now() time.Time {
    if a.Now != nil {
        return a.Now()
    }
    return time.Now()
}

// PostIfCooldownExpired implements PauseHintPoster.
func (a *DDBPauseHinter) PostIfCooldownExpired(ctx context.Context, channelID, threadTS string) error {
    info, err := a.SandboxByChannel.FetchByChannel(ctx, channelID)
    if err != nil {
        return fmt.Errorf("pause-hint: channel lookup: %w", err)
    }
    if info.SandboxID == "" {
        return nil // unknown channel — nothing to do
    }

    // Read last_pause_hint_ts via GetItem on km-sandboxes/{sandbox_id}.
    // (Phase 63 already grants GetItem on km-sandboxes; this is reuse.)
    out, err := a.Client.GetItem(ctx, &dynamodb.GetItemInput{
        TableName: awssdk.String(a.SandboxesTableName),
        Key: map[string]ddbtypes.AttributeValue{
            "sandbox_id": &ddbtypes.AttributeValueMemberS{Value: info.SandboxID},
        },
        ProjectionExpression: awssdk.String("last_pause_hint_ts"),
    })
    if err != nil {
        return fmt.Errorf("pause-hint: get last_pause_hint_ts: %w", err)
    }

    nowUnix := a.now().Unix()
    lastPostedSec := int64(0)
    if v, ok := out.Item["last_pause_hint_ts"]; ok {
        if n, ok := v.(*ddbtypes.AttributeValueMemberN); ok {
            // best-effort parse; absent or malformed → 0 → cooldown expired
            fmt.Sscanf(n.Value, "%d", &lastPostedSec)
        }
    }
    cooldown := a.CooldownSeconds
    if cooldown <= 0 {
        cooldown = 3600
    }
    if lastPostedSec != 0 && (nowUnix-lastPostedSec) <= cooldown {
        // Cooldown active — silently skip.
        return nil
    }

    // Cooldown expired or never set: conditional UpdateItem to absorb the
    // cold-start race.  Condition: attribute_not_exists OR last_pause_hint_ts <= lastSeenValue.
    // If the condition fails, another concurrent invocation already won and
    // posted — we DO NOT also post.
    nowStr := strconv.FormatInt(nowUnix, 10)
    cond := "attribute_not_exists(last_pause_hint_ts) OR last_pause_hint_ts <= :lastSeen"
    _, err = a.Client.UpdateItem(ctx, &dynamodb.UpdateItemInput{
        TableName: awssdk.String(a.SandboxesTableName),
        Key: map[string]ddbtypes.AttributeValue{
            "sandbox_id": &ddbtypes.AttributeValueMemberS{Value: info.SandboxID},
        },
        UpdateExpression:    awssdk.String("SET last_pause_hint_ts = :now"),
        ConditionExpression: awssdk.String(cond),
        ExpressionAttributeValues: map[string]ddbtypes.AttributeValue{
            ":now":      &ddbtypes.AttributeValueMemberN{Value: nowStr},
            ":lastSeen": &ddbtypes.AttributeValueMemberN{Value: strconv.FormatInt(lastPostedSec, 10)},
        },
    })
    if err != nil {
        var ccfe *ddbtypes.ConditionalCheckFailedException
        if errors.As(err, &ccfe) {
            // Lost the race — another invocation just won.  Don'''t double-post.
            return nil
        }
        return fmt.Errorf("pause-hint: write last_pause_hint_ts: %w", err)
    }

    // Won the race — post the hint.
    if err := a.Post(ctx, channelID, threadTS, a.HintText); err != nil {
        return fmt.Errorf("pause-hint: post: %w", err)
    }
    return nil
}
