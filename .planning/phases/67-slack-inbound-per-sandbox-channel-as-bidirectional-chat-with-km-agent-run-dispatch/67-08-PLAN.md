```go
func countActiveThreads(ctx context.Context, ddb DDBQueryClient, tableName, channelID string) (int, error) {
    if channelID == "" { return 0, nil }
    out, err := ddb.Query(ctx, &dynamodb.QueryInput{
        TableName:              awssdk.String(tableName),
        KeyConditionExpression: awssdk.String("channel_id = :cid"),
        ExpressionAttributeValues: map[string]ddbtypes.AttributeValue{
            ":cid": &ddbtypes.AttributeValueMemberS{Value: channelID},
        },
        Select: ddbtypes.SelectCount,
    })
    if err != nil { return 0, err }
    return int(out.Count), nil
}
```

```go
func checkSlackInboundQueueExists(ctx context.Context, deps doctorDeps) (status, msg string, err error) {
    // Query all km-sandboxes rows with non-empty slack_inbound_queue_url
    // (use a Scan with FilterExpression — sandbox count is small).
    rows, err := deps.ListSandboxesWithInbound(ctx)
    if err != nil { return "FAIL", "list sandboxes failed", err }
    if len(rows) == 0 { return "OK", "no sandboxes have inbound enabled", nil }

    var missing []string
    for _, r := range rows {
        if _, err := awspkg.QueueDepth(ctx, deps.SQS, r.QueueURL); err != nil {
            missing = append(missing, fmt.Sprintf("%s (%s)", r.SandboxID, r.QueueURL))
        }
    }
    if len(missing) > 0 {
        return "FAIL", fmt.Sprintf("inbound queue missing or unreachable for: %s", strings.Join(missing, ", ")), nil
    }
    return "OK", fmt.Sprintf("%d sandbox(es) have healthy inbound queues", len(rows)), nil
}

func checkSlackInboundStaleQueues(ctx context.Context, deps doctorDeps) (status, msg string, err error) {
    // List all queues matching {prefix}-slack-inbound-*.fifo
    prefix := deps.Cfg.GetResourcePrefix() + "-slack-inbound-"
    list, err := deps.SQS.ListQueues(ctx, &sqs.ListQueuesInput{
        QueueNamePrefix: awssdk.String(prefix),
    })
    if err != nil { return "FAIL", "list queues failed", err }
    if len(list.QueueUrls) == 0 { return "OK", "no inbound queues exist", nil }

    // Build set of known queue URLs from DDB
    rows, err := deps.ListSandboxesWithInbound(ctx)
    if err != nil { return "FAIL", "list sandboxes failed", err }
    known := map[string]string{}
    for _, r := range rows {
        known[r.QueueURL] = r.SandboxID
    }

    var stale []string
    for _, qURL := range list.QueueUrls {
        if _, ok := known[qURL]; !ok {
            stale = append(stale, qURL)
        }
    }
    if len(stale) > 0 {
        return "WARN", fmt.Sprintf("%d stale inbound queue(s) without DDB record: %s (run aws sqs delete-queue to clean)", len(stale), strings.Join(stale, ", ")), nil
    }
    return "OK", fmt.Sprintf("all %d inbound queue(s) accounted for in DDB", len(list.QueueUrls)), nil
}

func checkSlackAppEventsScopes(ctx context.Context, deps doctorDeps) (status, msg string, err error) {
    // Reuse the existing bridge token + auth.test path.
    // auth.test response includes a list of scopes the bot has.
    scopes, err := deps.SlackAuthTestScopes(ctx)
    if err != nil { return "WARN", "could not check Slack scopes", err }
    required := []string{"channels:history", "groups:history"}
    var missing []string
    for _, r := range required {
        found := false
        for _, s := range scopes {
            if s == r { found = true; break }
        }
        if !found { missing = append(missing, r) }
    }
    if len(missing) > 0 {
        return "FAIL",
            fmt.Sprintf("Slack App missing required scopes for inbound: %s. Add via Slack App config → OAuth & Permissions → Bot Token Scopes, then `km slack rotate-token`.", strings.Join(missing, ", ")),
            nil
    }
    return "OK", "Slack App has required inbound scopes", nil
}
```
