cmd.Flags().BoolVar(&enableTLS, "tls", false, "Enable TLS uprobe observability (attaches to libssl.so.3)")
cmd.Flags().StringVar(&allowedRepos, "allowed-repos", "", "Comma-separated list of allowed GitHub repos (owner/repo)")

In the `runEbpfAttach` function (or its equivalent), after the existing network enforcement setup, add TLS uprobe setup when `--tls` is true:

```go
if enableTLS {
    // 1. Discover libssl.so.3
    libsslPath, err := tls.FindSystemLibssl()
    if err != nil {
        log.Warn().Err(err).Msg("libssl.so.3 not found, skipping TLS uprobe")
    } else {
        // 2. Attach OpenSSL probes
        tlsProbe, err := tls.AttachOpenSSL(libsslPath)
        if err != nil {
            log.Error().Err(err).Msg("failed to attach OpenSSL uprobes")
        } else {
            defer tlsProbe.Close()
            log.Info().Str("libssl", libsslPath).Msg("OpenSSL uprobes attached")

            // 3. Create consumer
            consumer, err := tls.NewConsumer(tlsProbe.EventsMap(), log.Logger)
            if err != nil {
                log.Error().Err(err).Msg("failed to create TLS consumer")
            } else {
                // 4. Register handlers
                repos := strings.Split(allowedRepos, ",")
                ghHandler := tls.NewGitHubAuditHandler(repos, log.Logger)
                consumer.AddHandler(ghHandler.Handle)

                // 5. Run consumer in background goroutine
                tlsCtx, tlsCancel := context.WithCancel(ctx)
                defer tlsCancel()
                go func() {
                    if err := consumer.Run(tlsCtx); err != nil {
                        log.Error().Err(err).Msg("TLS consumer error")
                    }
                }()

                // Consumer will be stopped via context cancel on SIGTERM
                defer consumer.Close()
            }
        }
    }
}
```
