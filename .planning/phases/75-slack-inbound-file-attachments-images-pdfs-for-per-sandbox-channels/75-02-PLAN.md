       func sanitizeFilename(name string) string {
           // 1. Replace .. with _ (any literal ".." substring)
           name = strings.ReplaceAll(name, "..", "_")
           // 2. Replace / and \ with _
           name = strings.ReplaceAll(name, "/", "_")
           name = strings.ReplaceAll(name, "\\", "_")
           // 3. Strip non-printable bytes (rune-aware; strconv.IsPrint or unicode.IsPrint)
           var b strings.Builder
           for _, r := range name {
               if unicode.IsPrint(r) && r != ' ' {
                   b.WriteRune(r)
               }
           }
           name = b.String()
           // 4. Truncate to 255 bytes, rune-aligned
           if len(name) > 255 {
               name = truncateRunesToByteLimit(name, 255)
           }
           // 5. Empty result → "_"
           if name == "" {
               name = "_"
           }
           return name
       }
