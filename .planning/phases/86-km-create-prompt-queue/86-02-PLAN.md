       ```go
       func resolvePrompts(raw []string) ([]string, error) {
           out := make([]string, len(raw))
           for i, v := range raw {
               switch {
               case strings.HasPrefix(v, "@@"):
                   out[i] = v[1:] // @@x -> literal @x
               case strings.HasPrefix(v, "@"):
                   path := v[1:]
                   data, err := os.ReadFile(path)
                   if err != nil {
                       return nil, fmt.Errorf("--prompt @%s: %w", path, err)
                   }
                   out[i] = string(data) // UTF-8 verbatim
               default:
                   out[i] = v
               }
           }
           return out, nil
       }
       ```
       Note: error message MUST include the path so operators can debug. The wrapped error from `os.ReadFile` carries the syscall detail.

    2. **`pushQueueFiles(ctx, ssmClient SSMSendAPI, instanceID string, prompts []string, noBedrock bool) error`** — builds ONE shell command string and dispatches via `sendSSMAndWait`. Pattern (from RESEARCH.md `SSM Batch Queue File Push`):
       ```go
       var sb strings.Builder
       sb.WriteString("set -eu\n")
       sb.WriteString("mkdir -p /workspace/.km-agent/queue\n")
       sb.WriteString("chmod 0700 /workspace/.km-agent/queue\n")
       createdAt := time.Now().UTC().Format(time.RFC3339)
       noBedrockVal := "false"
       if noBedrock { noBedrockVal = "true" }
       for i, p := range prompts {
           num := fmt.Sprintf("%03d", i+1) // 001, 002, ...
           b64 := base64.StdEncoding.EncodeToString([]byte(p))
           // Prompt file
           sb.WriteString(fmt.Sprintf("echo %s | base64 -d > /workspace/.km-agent/queue/%s.prompt\n", b64, num))
           sb.WriteString(fmt.Sprintf("chmod 0600 /workspace/.km-agent/queue/%s.prompt\n", num))
           // Meta JSON
           meta := fmt.Sprintf(`{"no_bedrock":%s,"created_at":"%s","status":"pending"}`, noBedrockVal, createdAt)
           metaB64 := base64.StdEncoding.EncodeToString([]byte(meta))
           sb.WriteString(fmt.Sprintf("echo %s | base64 -d > /workspace/.km-agent/queue/%s.meta.json\n", metaB64, num))
           sb.WriteString(fmt.Sprintf("chmod 0600 /workspace/.km-agent/queue/%s.meta.json\n", num))
       }
       sb.WriteString("chown -R sandbox:sandbox /workspace/.km-agent/queue\n")
       _, err := sendSSMAndWait(ctx, ssmClient, instanceID, sb.String())
       return err
       ```
       Use `base64.StdEncoding` for both prompt AND meta to avoid heredoc-nesting / shell-escaping issues. One atomic SendCommand keeps the operator's mental model simple.
