// ExitCodeError is a typed sentinel error carrying a numeric process exit code.
// The outermost cobra command boundary (post-Execute, NOT inside RunE) detects this
// via errors.As and translates it into os.Exit(Code). This preserves any deferred
// cleanup registered by RunE / Cobra middleware (which inline os.Exit would skip).
type ExitCodeError struct {
    Code  int   // process exit code to propagate
    Inner error // optional wrapped underlying error
}

func (e *ExitCodeError) Error() string { /* "queue chain failed (exit code N): <inner>" */ }
func (e *ExitCodeError) Unwrap() error  { return e.Inner }

// waitForQueueDrain polls /workspace/.km-agent/queue/*.meta.json every 5s until

//   - (>0, nil)       first failed entry's exit code (read from /workspace/.km-agent/runs/<runID>/exit_code).
//   - (-1, err)       SSM/IO error during polling OR ctx.Err() returned.
// Respects ctx — exits promptly on cancellation.
func waitForQueueDrain(ctx context.Context, ssmClient SSMSendAPI, instanceID string, expectedCount int) (exitCode int, err error)

    1. **Add `ExitCodeError` to `create_prompt.go`:**
       ```go
       // ExitCodeError carries a process exit code up the cobra call stack.
       // The outermost command boundary (RunE-wrapper in NewCreateCmd) detects
       // this via errors.As and translates it into os.Exit(Code) AT the outermost
       // boundary — never inside RunE — so any deferred cleanup registered
       // by RunE or Cobra middleware runs to completion before the process exits.
       type ExitCodeError struct {
           Code  int
           Inner error
       }

       func (e *ExitCodeError) Error() string {
           if e.Inner != nil {
               return fmt.Sprintf("queue chain failed (exit code %d): %v", e.Code, e.Inner)
           }
           return fmt.Sprintf("queue chain failed (exit code %d)", e.Code)
       }

       func (e *ExitCodeError) Unwrap() error { return e.Inner }
       ```

       Why a typed error (not inline `os.Exit`):
       - Inline `os.Exit` inside `doStep16PromptPush` bypasses any `defer` statements registered higher in the call stack (Cobra's RunE wrappers, signal handlers, telemetry flushes, AWS SDK connection cleanup).
       - A typed error round-trips through `RunE`'s `error` return value cleanly; Cobra's normal error path runs all defers; the outermost layer (just outside `cmd.Execute()`) inspects with `errors.As` and only then calls `os.Exit(code)`.
       - This matches Cobra-community idioms (see e.g. kubectl's `cmdutil.CheckErr` + typed errors pattern).

    2. **Add `waitForQueueDrain` to `create_prompt.go`:**
       ```go
       var queuePollInterval = 5 * time.Second // var, not const, for test override

       // waitForQueueDrain blocks until every queue entry is terminal.
       // Returns 0 on all-done, the first-failed entry's exit code otherwise.
       // Respects ctx.Done() — returns ctx.Err() promptly on cancellation.
       func waitForQueueDrain(ctx context.Context, ssmClient SSMSendAPI, instanceID string, expectedCount int) (int, error) {
           statusCmd := `sudo -u sandbox bash -c '
       for f in $(ls -1 /workspace/.km-agent/queue/*.meta.json 2>/dev/null | sort); do
           [ -f "$f" ] || continue
           num=$(basename "$f" .meta.json)
           status=$(jq -r .status "$f" 2>/dev/null || echo "unknown")
           echo "$num|$status"
       done'`

           ticker := time.NewTicker(queuePollInterval)
           defer ticker.Stop()
           // First poll immediately (don't wait 5s for the first read)
           timer := time.NewTimer(0)
           defer timer.Stop()

           for {
               select {
               case <-ctx.Done():
                   return -1, ctx.Err()
               case <-timer.C:
                   // first iteration only; subsequent rely on ticker
               case <-ticker.C:
               }

               out, err := sendSSMAndWait(ctx, ssmClient, instanceID, statusCmd)
               if err != nil {
                   return -1, fmt.Errorf("queue status poll: %w", err)
               }

               statuses := parseQueueStatuses(out)
               if len(statuses) < expectedCount {
                   continue
               }

               terminal := true
               var firstFailedNum string
               for _, s := range statuses {
                   switch s.status {
                   case "done", "skipped":
                       // terminal-ok
                   case "failed":
                       if firstFailedNum == "" {
                           firstFailedNum = s.num
                       }
                   case "pending", "running":
                       terminal = false
                   default:
                       terminal = false
                   }
               }
               if !terminal {
                   continue
               }

               if firstFailedNum == "" {
                   return 0, nil
               }
               exitCode, _ := fetchFailedExitCode(ctx, ssmClient, instanceID)
               if exitCode == 0 {
                   exitCode = 1 // safety: ensure non-zero on failure
               }
               return exitCode, nil
           }
       }

       type queueStatusEntry struct {
           num    string
           status string
       }

       func parseQueueStatuses(out string) []queueStatusEntry {
           var entries []queueStatusEntry
           for _, line := range strings.Split(strings.TrimSpace(out), "\n") {
               line = strings.TrimSpace(line)
               if line == "" { continue }
               parts := strings.SplitN(line, "|", 2)
               if len(parts) != 2 { continue }
               entries = append(entries, queueStatusEntry{num: parts[0], status: parts[1]})
           }
           return entries
       }

       func fetchFailedExitCode(ctx context.Context, ssmClient SSMSendAPI, instanceID string) (int, error) {
           cmd := `sudo -u sandbox bash -c '
       for d in $(ls -1t /workspace/.km-agent/runs/ 2>/dev/null); do
           s=$(cat "/workspace/.km-agent/runs/$d/status" 2>/dev/null)
           if [ "$s" = "failed" ]; then
               cat "/workspace/.km-agent/runs/$d/exit_code" 2>/dev/null
               exit 0
           fi
       done
       echo 1'`
           out, err := sendSSMAndWait(ctx, ssmClient, instanceID, cmd)
           if err != nil { return 1, err }
           ec, parseErr := strconv.Atoi(strings.TrimSpace(out))
           if parseErr != nil { return 1, parseErr }
           return ec, nil
       }
       ```
