    # ============================================================
    # 6. iptables DNAT: redirect DNS and HTTP traffic to sidecars
    # ============================================================
    echo "[km-bootstrap] Configuring iptables DNAT..."
    # Redirect all outbound DNS (UDP 53) to local DNS proxy (port 53)
    iptables -t nat -A OUTPUT -p udp --dport 53 -j REDIRECT --to-ports 53
    iptables -t nat -A OUTPUT -p tcp --dport 53 -j REDIRECT --to-ports 53
    # Redirect all outbound HTTP (80) and HTTPS (443) to HTTP proxy (port 3128)
    # Exclude proxy process itself to avoid loop (by UID of km-proxy user)
    useradd -r km-proxy || true
    iptables -t nat -A OUTPUT -p tcp --dport 80 -m owner ! --uid-owner km-proxy -j REDIRECT --to-ports 3128
