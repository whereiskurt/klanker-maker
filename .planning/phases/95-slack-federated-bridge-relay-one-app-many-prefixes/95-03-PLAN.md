```go
func checkSlackPeerBridges(peerBridges []string, ownBridgeURL string) CheckResult {
    name := "Slack peer bridges"
    if len(peerBridges) == 0 {
        return CheckResult{Name: name, Status: CheckSkipped,
            Message: "slack.peer_bridges not set — federation off"}
    }
    // For each URL: url.Parse error => WARN (malformed); URL == ownBridgeURL => WARN (self-loop).
    // (Empty-on-front-door is the len==0 case above when the install hosts the Request URL — a
    //  WARN-vs-SKIP nuance the planner can resolve: if ownBridgeURL is the Slack Request URL host
    //  and peerBridges is empty, prefer WARN. Detecting "is front door" reliably is hard, so the
    //  pragmatic check is: malformed + self-loop are WARN; empty is SKIP unless a clear front-door
    //  signal exists. Match the design spec's intent; do not over-engineer host detection.)
    // All valid + no self-loop => CheckOK.
}
```
