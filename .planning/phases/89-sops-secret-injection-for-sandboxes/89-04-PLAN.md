       ```go
       func deleteOwnSecretsKMSAlias(ctx context.Context, client KMSAliasDeleter, resourcePrefix string) error {
           wantName := fmt.Sprintf("alias/%s-sandbox-secrets", resourcePrefix)
           var marker *string
           var keyID string
           found := false
           for {
               out, err := client.ListAliases(ctx, &kms.ListAliasesInput{Marker: marker})
               if err != nil {
                   return fmt.Errorf("list aliases: %w", err)
               }
               for _, a := range out.Aliases {
                   name := aws.ToString(a.AliasName)
                   if name == wantName {
                       found = true
                       keyID = aws.ToString(a.TargetKeyId)
                       break
                   }
               }
               if found || !out.Truncated {
                   break
               }
               marker = out.NextMarker
           }
           if !found {
               log.Info().Str("resource_prefix", resourcePrefix).Msg("no own sandbox-secrets alias found — skipping KMS cleanup")
               return nil
           }
           // Delete the alias first (cosmetic), then schedule key deletion (7-day window).
           if _, err := client.DeleteAlias(ctx, &kms.DeleteAliasInput{AliasName: aws.String(wantName)}); err != nil {
               return fmt.Errorf("delete alias %s: %w", wantName, err)
           }
           pendingDays := int32(7)
           if _, err := client.ScheduleKeyDeletion(ctx, &kms.ScheduleKeyDeletionInput{
               KeyId:              aws.String(keyID),
               PendingWindowInDays: &pendingDays,
           }); err != nil {
               return fmt.Errorf("schedule key deletion %s: %w", keyID, err)
           }
           log.Info().Str("alias", wantName).Str("key_id", keyID).Int32("pending_days", pendingDays).Msg("scheduled sandbox-secrets KMS key deletion")
           return nil
       }
       ```
