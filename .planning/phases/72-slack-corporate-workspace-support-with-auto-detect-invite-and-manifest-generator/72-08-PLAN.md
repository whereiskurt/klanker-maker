```go
func checkSlackFilesWriteScope(ctx context.Context, deps *DoctorDeps) DoctorCheckResult {
    botToken, err := deps.SSM.Get(ctx, deps.SsmPrefix+"slack/bot-token", true)
    if err != nil || botToken == "" {
        return DoctorCheckResult{Status: StatusSkip, Message: "Slack not configured"}
    }
    scopes, err := fetchSlackBotScopes(ctx, botToken)
    if err != nil {
        return DoctorCheckResult{Status: StatusError, Message: fmt.Sprintf("fetch scopes: %v", err)}
    }
    for _, s := range scopes {
        if s == "files:write" { return DoctorCheckResult{Status: StatusOK} }
    }
    return DoctorCheckResult{
        Status: StatusWarn,
        Message: "Slack bot is missing files:write scope. Run `km slack manifest` to see required scopes; update Slack App config; reinstall; rotate token.",
    }
}
```

```go
// checkSlackUsersReadEmailScope verifies the bot has the users:read.email
// scope required by the EnsureMemberByEmail orchestrator (Phase 72). Without
// this scope, `km slack invite` and `km create` auto-invite both fail with
// cryptic `missing_scope` errors. This check pre-empts that by surfacing
// scope drift at `km doctor` time.
//
// Status semantics:
//   OK    — scope present
//   WARN  — scope missing (with remediation pointing at `km slack manifest`)
//   SKIP  — Slack not configured (no bot token in SSM)
//   ERROR — could not fetch scopes (network / token / API issue)
func checkSlackUsersReadEmailScope(ctx context.Context, deps *DoctorDeps) DoctorCheckResult {
    botToken, err := deps.SSM.Get(ctx, deps.SsmPrefix+"slack/bot-token", true)
    if err != nil || botToken == "" {
        return DoctorCheckResult{Status: StatusSkip, Message: "Slack not configured"}
    }
    scopes, err := fetchSlackBotScopes(ctx, botToken)
    if err != nil {
        return DoctorCheckResult{
            Status:  StatusError,
            Message: fmt.Sprintf("fetch Slack bot scopes: %v", err),
        }
    }
    for _, s := range scopes {
        if s == "users:read.email" {
            return DoctorCheckResult{Status: StatusOK}
        }
    }
    return DoctorCheckResult{
        Status: StatusWarn,
        Message: "Slack bot is missing the `users:read.email` scope (required by `km slack invite` and the auto-invite feature in `km create`). Remediation: run `km slack manifest > app.json`, update the Slack App's bot scopes from app.json (Slack Admin → Apps → your app → OAuth & Permissions → Bot Token Scopes), reinstall the app, then `km slack rotate-token --bot-token <new-token>`.",
    }
}
```
