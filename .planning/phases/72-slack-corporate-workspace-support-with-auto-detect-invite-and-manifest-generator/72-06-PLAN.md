    switch res {
    case slack.InvitedDirect:

    case slack.InvitedConnect:
        log.Info().Str("email", inv).Str("channel", chID).Msg("Sent Slack Connect invite to operator")
    case slack.AlreadyMember:
        log.Info().Str("email", inv).Str("channel", chID).Msg("Operator is already a member of Slack channel")
    case slack.SkippedExternal:
        log.Warn().Str("email", inv).Str("channel", chID).Msg("Operator email is not a workspace member; skipped Connect invite (run `km slack invite --external` to send one)")

    }
}
