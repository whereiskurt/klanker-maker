<objective>
Build the operator-side `km slack` command tree — `init`, `test`, `status` — that bootstraps the workspace integration, validates end-to-end delivery, and reports current state. `init` is the one-time setup operators run after building the binary; it requires a Pro Slack workspace and a bot token captured from the Slack App admin UI (out-of-band per CONTEXT.md).

The init flow is interactive (prompts for the bot token + invite email unless flags are passed) and idempotent: re-running it with the same workspace updates the token and re-deploys the Lambda but does not recreate the shared channel if it already exists in SSM (unless `--force` is passed). The flow makes Slack API calls directly from the operator's workstation — the bridge Lambda isn't a hard dependency at this stage because at bootstrap time it doesn't exist yet (the init flow is what deploys it).

Per RESEARCH.md Open Question #2, Lambda deployment is triggered by `km slack init` directly via `runner.Apply()` on `infra/live/{region}/lambda-slack-bridge/` (and the `dynamodb-slack-nonces` table that it depends on). Standard `km init` also deploys these modules now (Plan 06 added them to `regionalModules()`), so a fresh operator running `km init` first then `km slack init` works seamlessly. `km slack init` checks for existing deployment via SSM `/km/slack/bridge-url` presence and skips Terraform apply if already present (unless `--force`).

// SlackAPI is a narrow Slack-Web-API interface for testability. Production
// implementation is *slack.Client; test implementation is an in-memory fake.
type SlackAPI interface {
    AuthTest(ctx context.Context) error
    CreateChannel(ctx context.Context, name string) (string, error)
    InviteShared(ctx context.Context, channelID, email string) error
}

type SSMParamStore interface {
    Get(ctx context.Context, name string, decrypt bool) (string, error)

type TerragruntRunner interface {

type Prompter interface {
    PromptString(label string) (string, error)

// SlackCmdDeps bundles the injectable dependencies — RunE wires production
// implementations; tests inject mocks.
type SlackCmdDeps struct {
    NewSlackAPI func(token string) SlackAPI
    SSM         SSMParamStore
    Terragrunt  TerragruntRunner
    Prompter    Prompter
    OperatorKeyLoader func(ctx context.Context, region string) (ed25519.PrivateKey, error)
    BridgePoster      func(ctx context.Context, bridgeURL string, env *slack.SlackEnvelope, sig []byte) (*slack.PostResponse, error)
    Region            string
}

func NewSlackCmd(cfg *config.Config) *cobra.Command {
    c := &cobra.Command{
        Use:   "slack",
        Short: "Manage Slack-notify integration (Phase 63)",
    }
    c.AddCommand(newSlackInitCmd(cfg))
    c.AddCommand(newSlackTestCmd(cfg))
    c.AddCommand(newSlackStatusCmd(cfg))
    return c
}

func newSlackInitCmd(cfg *config.Config) *cobra.Command {
    var (
        botToken      string
        inviteEmail   string
        sharedChannel string
        force         bool
    )
    c := &cobra.Command{
        Use:   "init",
        Short: "One-time bootstrap: provision shared channel, deploy bridge Lambda, write SSM params",
        RunE: func(cmd *cobra.Command, args []string) error {
            deps := buildSlackDeps(cfg)
            return runSlackInit(cmd.Context(), deps, slackInitOpts{
                BotToken: botToken, InviteEmail: inviteEmail,
                SharedChannel: sharedChannel, Force: force,
            })
        },
    }
    c.Flags().StringVar(&botToken, "bot-token", "", "Slack bot token (xoxb-...); skip prompt")
    c.Flags().StringVar(&inviteEmail, "invite-email", "", "Email to send Slack Connect invite")
    c.Flags().StringVar(&sharedChannel, "shared-channel", "km-notifications", "Shared channel name")
    c.Flags().BoolVar(&force, "force", false, "Recreate shared channel and re-apply Lambda even if existing")
    return c
}

type slackInitOpts struct {
    BotToken, InviteEmail, SharedChannel string
    Force bool
}

func runSlackInit(ctx context.Context, d *SlackCmdDeps, opts slackInitOpts) error {
    // 1. Bot token: prompt if missing
    token := opts.BotToken
    if token == "" {
        existing, _ := d.SSM.Get(ctx, "/km/slack/bot-token", true)
        if existing != "" && !opts.Force {
            token = existing
        } else {
            t, err := d.Prompter.PromptSecret("Slack bot token (xoxb-...): ")
            if err != nil { return err }
            token = t
        }
    }

    // 2. Validate via auth.test
    api := d.NewSlackAPI(token)
    if err := api.AuthTest(ctx); err != nil {
        return fmt.Errorf("invalid Slack bot token: %w", err)
    }

    // 3. Persist token
    if err := d.SSM.Put(ctx, "/km/slack/bot-token", token, true); err != nil {
        return err
    }
    // ... workspace metadata ...

    // 4. Invite email
    inv := opts.InviteEmail
    if inv == "" {
        existing, _ := d.SSM.Get(ctx, "/km/slack/invite-email", false)
        if existing != "" && !opts.Force {
            inv = existing
        } else {
            v, err := d.Prompter.PromptString("Invite email for Slack Connect: ")
            if err != nil { return err }
            inv = v
        }
    }
    if err := d.SSM.Put(ctx, "/km/slack/invite-email", inv, false); err != nil {
        return err
    }

    // 5. Shared channel
    chID, _ := d.SSM.Get(ctx, "/km/slack/shared-channel-id", false)
    if chID == "" || opts.Force {
        newID, err := api.CreateChannel(ctx, opts.SharedChannel)
        if err != nil {
            // Translate name_taken into a clearer error if needed
            return fmt.Errorf("create shared channel %q: %w", opts.SharedChannel, err)
        }
        chID = newID
        if err := d.SSM.Put(ctx, "/km/slack/shared-channel-id", chID, false); err != nil {
            return err
        }
        if err := api.InviteShared(ctx, chID, inv); err != nil {
            return fmt.Errorf("send Slack Connect invite to %s (workspace must be Pro): %w", inv, err)
        }
    }

    // 6. Lambda + nonce table deploy
    bridgeURL, _ := d.SSM.Get(ctx, "/km/slack/bridge-url", false)
    if bridgeURL == "" || opts.Force {
        for _, m := range []string{"dynamodb-slack-nonces", "lambda-slack-bridge"} {
            modulePath := fmt.Sprintf("infra/live/%s/%s", regionDir(d.Region), m)
            if err := d.Terragrunt.Apply(ctx, modulePath); err != nil {
                return fmt.Errorf("terragrunt apply %s: %w", m, err)
            }
        }
        url, err := d.Terragrunt.Output(ctx, fmt.Sprintf("infra/live/%s/lambda-slack-bridge", regionDir(d.Region)), "function_url")
        if err != nil {
            return fmt.Errorf("read function_url output: %w", err)
        }
        bridgeURL = url
        if err := d.SSM.Put(ctx, "/km/slack/bridge-url", bridgeURL, false); err != nil {
            return err
        }
    }

    fmt.Println("km slack init: complete.")
    fmt.Printf("  shared channel: %s\n", chID)
    fmt.Printf("  bridge URL:     %s\n", bridgeURL)
    fmt.Printf("  invite sent to: %s\n", inv)
    return nil
}

func runSlackTest(ctx context.Context, d *SlackCmdDeps) error {
    bridgeURL, err := d.SSM.Get(ctx, "/km/slack/bridge-url", false)
    if err != nil || bridgeURL == "" {
        return errors.New("/km/slack/bridge-url not set; run km slack init first")
    }
    chID, err := d.SSM.Get(ctx, "/km/slack/shared-channel-id", false)
    if err != nil || chID == "" {
        return errors.New("/km/slack/shared-channel-id not set; run km slack init first")
    }
    priv, err := d.OperatorKeyLoader(ctx, d.Region)
    if err != nil {
        return fmt.Errorf("load operator key: %w", err)
    }
    env, err := slack.BuildEnvelope(slack.ActionTest, slack.SenderOperator, chID, "km slack test", "If you see this, the bridge is wired.", "")
    if err != nil { return err }
    _, sig, err := slack.SignEnvelope(env, priv)
    if err != nil { return err }
    resp, err := d.BridgePoster(ctx, bridgeURL, env, sig)
    if err != nil { return err }
    if !resp.OK {
        return fmt.Errorf("bridge returned not-ok: %s", resp.Error)
    }
    fmt.Printf("km slack test: posted ts=%s\n", resp.TS)
    _ = d.SSM.Put(ctx, "/km/slack/last-test-timestamp", time.Now().UTC().Format(time.RFC3339), false)
    return nil
}

func runSlackStatus(ctx context.Context, d *SlackCmdDeps) error {

        if v == "" {

        }
        fmt.Printf("%-40s %s\n", k, v)
    }
    return nil
}
