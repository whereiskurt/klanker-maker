Recommendation: Option B (adapter does its own thin HTTP) — keeps pkg/slack.Client API small and avoids an awkward back-channel. The adapter posts to chat.postMessage / conversations.archive directly with the bot token, and inspects `resp.Header.Get("Retry-After")` on 429.

```go
type SlackPosterAdapter struct {
    Token      string
    HTTPClient *http.Client
}

func (s *SlackPosterAdapter) PostMessage(ctx context.Context, channel, subject, body, threadTS string) (string, error) {
    text := fmt.Sprintf("*%s*\n\n%s", subject, body)
    resp, err := s.call(ctx, "chat.postMessage", map[string]any{
        "channel": channel, "text": text,
        "unfurl_links": false, "unfurl_media": false, "mrkdwn": true,
        // include thread_ts only if non-empty
    })
    if err != nil { return "", err }
    if resp.RetryAfter > 0 {
        return "", &bridge.ErrSlackRateLimited{RetryAfterSeconds: resp.RetryAfter, Method: "chat.postMessage"}
    }
    if !resp.OK { return "", fmt.Errorf("chat.postMessage: %s", resp.Error) }
    return resp.TS, nil
}

// call returns a struct with OK, Error, TS, RetryAfter parsed from response + headers.
```

```go
// Command km-slack-bridge is the Phase 63 Slack-notify Lambda. It accepts
// signed envelopes from sandboxes and the operator and dispatches to the
// Slack Web API. See pkg/slack/bridge for the verification logic.
package main

import (
    "context"
    "encoding/json"
    "log"
    "net/http"
    "os"
    "time"

    "github.com/aws/aws-lambda-go/events"
    "github.com/aws/aws-lambda-go/lambda"
    awsconfig "github.com/aws/aws-sdk-go-v2/config"
    "github.com/aws/aws-sdk-go-v2/service/dynamodb"
    "github.com/aws/aws-sdk-go-v2/service/ssm"

    "yourrepo/pkg/slack/bridge"
)

var handler *bridge.Handler

func init() {
    ctx := context.Background()
    cfg, err := awsconfig.LoadDefaultConfig(ctx)
    if err != nil { log.Fatalf("load aws config: %v", err) }
    ddb := dynamodb.NewFromConfig(cfg)
    ssmc := ssm.NewFromConfig(cfg)

    identitiesTable := envOr("KM_IDENTITIES_TABLE", "km-identities")
    sandboxesTable  := envOr("KM_SANDBOXES_TABLE", "km-sandboxes")
    nonceTable      := envOr("KM_NONCE_TABLE", "km-slack-bridge-nonces")
    botTokenPath    := envOr("KM_BOT_TOKEN_PATH", "/km/slack/bot-token")

    keys := &bridge.DynamoPublicKeyFetcher{Client: ddb, TableName: identitiesTable}
    nonces := &bridge.DynamoNonceStore{Client: ddb, TableName: nonceTable}
    channels := &bridge.DynamoChannelOwnershipFetcher{Client: ddb, TableName: sandboxesTable}
    token := &bridge.SSMBotTokenFetcher{Client: ssmc, Path: botTokenPath}

    // SlackPosterAdapter constructs a token-aware poster lazily; it fetches
    // the bot token from `token` on the first call within a Lambda warm
    // container, then caches for 15 min.
    poster := &bridge.SlackPosterAdapter{HTTPClient: &http.Client{Timeout: 10 * time.Second}, Tokens: token}

    handler = &bridge.Handler{
        Now:      time.Now,
        Keys:     keys,
        Nonces:   nonces,
        Channels: channels,
        Token:    token,
        Slack:    poster,
    }
}

func main() {
    lambda.Start(handle)
}

func handle(ctx context.Context, ev events.LambdaFunctionURLRequest) (events.LambdaFunctionURLResponse, error) {
    req := &bridge.Request{
        Body:    ev.Body,
        Headers: ev.Headers,
    }
    resp := handler.Handle(ctx, req)
    body, _ := json.Marshal(json.RawMessage(resp.Body))
    _ = body
    return events.LambdaFunctionURLResponse{
        StatusCode: resp.StatusCode,
        Body:       resp.Body,
        Headers:    resp.Headers,
    }, nil
}

func envOr(k, def string) string {
    v := os.Getenv(k)
    if v == "" { return def }
    return v
}
```
