```go
package cmd

import (
    "context"
    "crypto/ed25519"
    "fmt"
    "log"
    "time"

    "yourrepo/pkg/aws"  // SandboxMetadata
    "yourrepo/pkg/slack"
)

// BridgePoster is the narrow interface for sending signed envelopes to the
// bridge Lambda. Production = slack.PostToBridge; test = mock.
type BridgePoster func(ctx context.Context, bridgeURL string, env *slack.SlackEnvelope, sig []byte) (*slack.PostResponse, error)

// OperatorKeyLoader fetches the operator's Ed25519 private key from SSM.
// Production reads /sandbox/operator/signing-key (Phase 14 / 45 helper).
type OperatorKeyLoader func(ctx context.Context, region string) (ed25519.PrivateKey, error)

// destroySlackChannel runs the per-sandbox Slack teardown matrix from CONTEXT.md.
// Always returns nil — Slack failures NEVER block km destroy. WARN-level logs
// surface skip / failure cases for the operator to review.
//
// Reads m.SlackArchiveOnDestroy directly from the metadata record (populated
// at create time by Plan 63-08) so destroy is self-contained.
func destroySlackChannel(ctx context.Context, m *aws.SandboxMetadata,
                         region string, ssm SSMParamStore, keyLoader OperatorKeyLoader,
                         poster BridgePoster) error {

    if m == nil || m.SlackChannelID == "" {
        return nil // Case A — no Slack
    }
    if !m.SlackPerSandbox {
        return nil // Case B — shared mode (never archive)
    }

    bridgeURL, _ := ssm.Get(ctx, "/km/slack/bridge-url", false)
    if bridgeURL == "" {
        log.Printf("WARN: km destroy: /km/slack/bridge-url not set; skipping Slack archive for channel %s (sandbox %s). Run km slack init or manually archive in Slack.", m.SlackChannelID, m.SandboxID)
        return nil
    }

    priv, err := keyLoader(ctx, region)
    if err != nil {
        log.Printf("WARN: km destroy: operator key load failed: %v; skipping Slack archive for channel %s.", err, m.SlackChannelID)
        return nil
    }

    // Determine archive intent: nil pointer = default true; non-nil = explicit value.
    archive := m.SlackArchiveOnDestroy == nil || *m.SlackArchiveOnDestroy

    // Final post.
    timestamp := time.Now().UTC().Format(time.RFC3339)
    body := fmt.Sprintf("Sandbox `%s` destroyed at %s.", m.SandboxID, timestamp)
    if !archive {
        body += "\n\nChannel preserved per profile.slackArchiveOnDestroy=false."
    }
    finalEnv, err := slack.BuildEnvelope(slack.ActionPost, slack.SenderOperator, m.SlackChannelID,
        fmt.Sprintf("[%s] destroyed", m.SandboxID), body, "")
    if err != nil {
        log.Printf("WARN: km destroy: build final envelope: %v", err)
        return nil
    }
    _, finalSig, err := slack.SignEnvelope(finalEnv, priv)
    if err != nil {
        log.Printf("WARN: km destroy: sign final envelope: %v", err)
        return nil
    }
    finalResp, err := poster(ctx, bridgeURL, finalEnv, finalSig)
    if err != nil || (finalResp != nil && !finalResp.OK) {
        log.Printf("WARN: km destroy: final post to channel %s failed: %v", m.SlackChannelID, err)
        // Don't attempt archive if final post failed.
        return nil
    }

    if !archive {
        return nil // Case E
    }

    // Archive call.
    archEnv, err := slack.BuildEnvelope(slack.ActionArchive, slack.SenderOperator, m.SlackChannelID, "", "", "")
    if err != nil {
        log.Printf("WARN: km destroy: build archive envelope: %v", err)
        return nil
    }
    _, archSig, err := slack.SignEnvelope(archEnv, priv)
    if err != nil {
        log.Printf("WARN: km destroy: sign archive envelope: %v", err)
        return nil
    }
    archResp, err := poster(ctx, bridgeURL, archEnv, archSig)
    if err != nil || (archResp != nil && !archResp.OK) {
        log.Printf("WARN: km destroy: archive of channel %s failed: %v (channel persists; manually archive via Slack UI)", m.SlackChannelID, err)
        return nil
    }
    log.Printf("km destroy: archived Slack channel %s for sandbox %s", m.SlackChannelID, m.SandboxID)
    return nil
}
```

```go
// checkSlackTokenValidity exercises the bridge Lambda end-to-end with an
// operator-signed action=test envelope. Confirms the bot token in SSM is
// valid AND the bridge Lambda is deployed, reachable, and able to call Slack.
func checkSlackTokenValidity(ctx context.Context, ssm SSMParamStore, region string,
                              keyLoader OperatorKeyLoader, poster BridgePoster) CheckResult {
    name := "slack-token"
    botToken, _ := ssm.Get(ctx, "/km/slack/bot-token", true)
    if botToken == "" {
        return CheckResult{Name: name, Status: "SKIPPED", Message: "Slack not configured (run km slack init to enable)"}
    }
    bridgeURL, _ := ssm.Get(ctx, "/km/slack/bridge-url", false)
    if bridgeURL == "" {
        return CheckResult{Name: name, Status: "WARN",
            Message: "Slack bot token configured but bridge Lambda not deployed",
            Remediation: "Run `km slack init` to deploy the bridge Lambda"}
    }
    sharedCh, _ := ssm.Get(ctx, "/km/slack/shared-channel-id", false)
    if sharedCh == "" {
        return CheckResult{Name: name, Status: "WARN",
            Message: "Shared channel not provisioned",
            Remediation: "Run `km slack init` to create the shared channel"}
    }
    priv, err := keyLoader(ctx, region)
    if err != nil {
        return CheckResult{Name: name, Status: "ERROR", Message: fmt.Sprintf("operator key load: %v", err)}
    }
    env, _ := slack.BuildEnvelope(slack.ActionTest, slack.SenderOperator, sharedCh, "km doctor", "auth check", "")
    _, sig, _ := slack.SignEnvelope(env, priv)
    resp, err := poster(ctx, bridgeURL, env, sig)
    if err != nil {
        return CheckResult{Name: name, Status: "ERROR", Message: fmt.Sprintf("bridge unreachable: %v", err)}
    }
    if !resp.OK {
        return CheckResult{Name: name, Status: "WARN",
            Message: fmt.Sprintf("bridge returned not-ok: %s (token may be invalid)", resp.Error),
            Remediation: "Rotate token via `km slack init --force --bot-token <new>`"}
    }
    return CheckResult{Name: name, Status: "OK", Message: fmt.Sprintf("Slack bridge OK (test ts=%s)", resp.TS)}
}
```

```go
func checkStaleSlackChannels(ctx context.Context, store SandboxStore, ec2 EC2InstanceLister) CheckResult {

        return CheckResult{Name: name, Status: "ERROR", Message: fmt.Sprintf("list sandboxes: %v", err)}
    }
    var stale []string
    for _, s := range sandboxes {
        if s.SlackChannelID == "" || !s.SlackPerSandbox {
            continue
        }
        if s.InstanceID == "" {
            continue
        }
        exists, err := ec2.InstanceExists(ctx, s.InstanceID)
        if err != nil || !exists {
            stale = append(stale, fmt.Sprintf("%s→%s", s.SandboxID, s.SlackChannelID))
        }
    }
    if len(stale) == 0 {
        return CheckResult{Name: name, Status: "OK", Message: "No stale Slack channels"}
    }
    return CheckResult{Name: name, Status: "WARN",
        Message: fmt.Sprintf("%d stale Slack channels: %s", len(stale), strings.Join(stale, ", ")),
        Remediation: "Archive manually via Slack UI, or write a one-off script using km email/internal cleanup helper"}
}
```
