# Locked gate algorithm (CONTEXT.md decisions lines 117-135)
trips := []Trip{}
for _, r := range reports {
    if r.ParseFailed {
        trips = append(trips, Trip{Module: r.Module, Action: "PARSE-FAIL", Reason: "plan JSON parse failed — conservative trip"})
        continue
    }
    for _, ch := range r.Destroys {
        if slices.Contains(ProtectedTypes, ch.Type) {
            trips = append(trips, Trip{Module: r.Module, Type: ch.Type, Address: ch.Address, Action: "DESTROY"})
        }
    }
    for _, ch := range r.Replaces {
        if slices.Contains(ProtectedTypes, ch.Type) {
            trips = append(trips, Trip{Module: r.Module, Type: ch.Type, Address: ch.Address, Action: "REPLACE"})
        }
    }
}
blocked := len(trips) > 0 && !acceptDestroys
return GateResult{Blocked: blocked, Trips: trips}

# Action classification (Terraform show -json schema, format_version 1.x)
# Authoritative: https://developer.hashicorp.com/terraform/internals/json-format
#   ["no-op"]                              → no-op
#   ["create"]                             → create
#   ["read"]                               → read (data source)
#   ["update"]                             → update
#   ["delete"]                             → delete
#   ["delete","create"] / ["create","delete"] → replace (either order)
