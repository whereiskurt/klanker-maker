    ```bash
    #!/bin/bash
    TOKEN=$(aws ssm get-parameter \
      --name "/sandbox/${SANDBOX_ID}/github-token" \
      --with-decryption \
      --query "Parameter.Value" \
      --output text 2>/dev/null || echo "")

      *Password*) echo "$TOKEN" ;;
      *)          echo "" ;;
    esac
    ```
