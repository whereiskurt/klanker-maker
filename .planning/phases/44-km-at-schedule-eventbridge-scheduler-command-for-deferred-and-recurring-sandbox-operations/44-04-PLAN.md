    **Test structure:** Single top-level `TestAtE2E` function that runs subtests sequentially (t.Run). The entire test has a 12-minute timeout via `go test -timeout 15m`. The test uses the `buildKM(t)` helper already defined in `validate_test.go` (same package, so it's available).

    **Helper function:** `runKM(t *testing.T, km string, args ...string) (stdout string, stderr string, err error)` -- executes km binary with args, captures stdout+stderr separately, returns all three. Log the full command and output on failure for debugging.

    **Helper function:** `waitForSandbox(t *testing.T, km string, sandboxID string, timeout time.Duration) bool` -- polls `km list --wide` every 15 seconds until sandboxID appears (returns true) or timeout expires (returns false). Looks for sandboxID in the output.

    **Helper function:** `waitForSandboxGone(t *testing.T, km string, sandboxID string, timeout time.Duration) bool` -- polls `km list --wide` every 15 seconds until sandboxID no longer appears (returns true) or timeout expires (returns false).

    **Helper function:** `waitForSandboxState(t *testing.T, km string, sandboxID string, wantState string, timeout time.Duration) bool` -- polls `km list --wide` every 15 seconds until the sandboxID row contains wantState (e.g., "paused", "running", "stopped"). Returns true when matched.

    **Revised complete test flow:**

    ```go
    func TestAtE2E(t *testing.T) {
        if os.Getenv("KM_E2E") == "" {
            t.Skip("set KM_E2E=1 to run E2E tests")
        }

        km := buildKM(t)
        state := &e2eState{
            km:     km,
            testID: fmt.Sprintf("e2e-%s", time.Now().Format("0102-150405")),
        }

        // Safety cleanup: destroy any sandbox we created if test fails
        t.Cleanup(func() {
            if state.sandboxID != "" {
                runKM(t, km, "destroy", state.sandboxID, "--remote", "--yes")
            }
        })

        // Capture existing sandbox IDs for diff later
        preList, _, _ := runKM(t, km, "list", "--wide")
        existingIDs := extractSandboxIDs(preList)

        // --- Step 1: Schedule one-time create 2 min from now ---
        t.Run("01-schedule-create", func(t *testing.T) {
            name := state.testID + "-create"
            stdout, stderr, err := runKM(t, km, "at", "in 2 minutes", "create", "profiles/sealed.yaml", "--remote", "--name", name)
            require.NoError(t, err, "stdout=%s stderr=%s", stdout, stderr)
            require.Contains(t, stdout, "Scheduled:")
        })

        // --- Step 2: Verify schedule appears in km at list ---
        t.Run("02-at-list-shows-schedule", func(t *testing.T) {
            stdout, _, err := runKM(t, km, "at", "list")
            require.NoError(t, err)
            require.Contains(t, stdout, state.testID+"-create")
        })

        // --- Step 3: Wait for scheduled create to fire (up to 4 min) ---
        t.Run("03-wait-sandbox-created", func(t *testing.T) {
            var found string
            deadline := time.Now().Add(4 * time.Minute)
            for time.Now().Before(deadline) {
                stdout, _, _ := runKM(t, km, "list", "--wide")
                newIDs := extractSandboxIDs(stdout)
                for _, id := range newIDs {
                    if !contains(existingIDs, id) {
                        found = id
                        break
                    }
                }
                if found != "" {
                    break
                }
                time.Sleep(15 * time.Second)
            }
            require.NotEmpty(t, found, "scheduled create did not produce a sandbox within 4 min")
            state.sandboxID = found
            t.Logf("scheduled create produced sandbox: %s", found)
        })

        // --- Step 4: Schedule one-time pause 1 min from now ---
        t.Run("04-schedule-pause", func(t *testing.T) {
            name := state.testID + "-pause"
            stdout, stderr, err := runKM(t, km, "at", "in 1 minute", "pause", state.sandboxID, "--name", name)
            require.NoError(t, err, "stdout=%s stderr=%s", stdout, stderr)
            require.Contains(t, stdout, "Scheduled:")
        })

        // --- Step 5: Wait for pause to take effect ---
        t.Run("05-wait-paused", func(t *testing.T) {
            ok := waitForSandboxState(t, km, state.sandboxID, "stopped", 3*time.Minute)
            require.True(t, ok, "sandbox %s did not reach stopped/paused state", state.sandboxID)
        })

        // --- Step 6: Schedule one-time resume 1 min from now ---
        t.Run("06-schedule-resume", func(t *testing.T) {
            name := state.testID + "-resume"
            stdout, stderr, err := runKM(t, km, "at", "in 1 minute", "resume", state.sandboxID, "--name", name)
            require.NoError(t, err, "stdout=%s stderr=%s", stdout, stderr)
        })

        // --- Step 7: Wait for resume ---
        t.Run("07-wait-resumed", func(t *testing.T) {
            ok := waitForSandboxState(t, km, state.sandboxID, "running", 3*time.Minute)
            require.True(t, ok, "sandbox %s did not resume to running state", state.sandboxID)
        })

        // --- Step 8: Schedule one-time extend (test that extend command dispatches) ---
        t.Run("08-schedule-extend", func(t *testing.T) {
            name := state.testID + "-extend"
            stdout, stderr, err := runKM(t, km, "at", "in 1 minute", "extend", state.sandboxID, "--name", name)
            require.NoError(t, err, "stdout=%s stderr=%s", stdout, stderr)
            // Extend just pushes the TTL forward -- we verify it scheduled without error
            // and fires without error (check km at list status after it fires)
        })

        // --- Step 9: Test km at cancel ---
        t.Run("09-at-cancel", func(t *testing.T) {
            // Schedule something, then cancel it before it fires
            cancelName := state.testID + "-cancel-test"
            stdout, stderr, err := runKM(t, km, "at", "in 10 minutes", "stop", state.sandboxID, "--name", cancelName)
            require.NoError(t, err, "stdout=%s stderr=%s", stdout, stderr)

            // Verify it appears in list
            listOut, _, _ := runKM(t, km, "at", "list")
            require.Contains(t, listOut, cancelName)

            // Cancel it
            stdout, stderr, err = runKM(t, km, "at", "cancel", cancelName)
            require.NoError(t, err, "stdout=%s stderr=%s", stdout, stderr)
            require.Contains(t, stdout, "Cancelled:")

            // Verify it's gone from list
            listOut, _, _ = runKM(t, km, "at", "list")
            require.NotContains(t, listOut, cancelName)
        })

        // --- Step 10: Test --cron flag with recurring schedule (create + cancel) ---
        t.Run("10-cron-recurring", func(t *testing.T) {
            cronName := state.testID + "-cron"
            // Schedule a recurring kill that we'll immediately cancel (we just want to verify creation works)
            stdout, stderr, err := runKM(t, km, "at", "--cron", "cron(0 0 1 1 ? 2099)", "kill", state.sandboxID, "--remote", "--name", cronName)
            require.NoError(t, err, "stdout=%s stderr=%s", stdout, stderr)
            require.Contains(t, stdout, "Scheduled:")

            // Verify in list with recurring indicator
            listOut, _, _ := runKM(t, km, "at", "list")
            require.Contains(t, listOut, cronName)

            // Cancel the recurring schedule
            stdout, stderr, err = runKM(t, km, "at", "cancel", cronName)
            require.NoError(t, err, "stdout=%s stderr=%s", stdout, stderr)

            // Verify gone
            listOut, _, _ = runKM(t, km, "at", "list")
            require.NotContains(t, listOut, cronName)
        })

        // --- Step 11: Schedule one-time kill to clean up ---
        t.Run("11-schedule-kill", func(t *testing.T) {
            name := state.testID + "-kill"
            stdout, stderr, err := runKM(t, km, "at", "in 1 minute", "kill", state.sandboxID, "--remote", "--name", name)
            require.NoError(t, err, "stdout=%s stderr=%s", stdout, stderr)
            require.Contains(t, stdout, "Scheduled:")
        })

        // --- Step 12: Wait for sandbox to be destroyed ---
        t.Run("12-wait-destroyed", func(t *testing.T) {
            ok := waitForSandboxGone(t, km, state.sandboxID, 4*time.Minute)
            require.True(t, ok, "sandbox %s was not destroyed within 4 min", state.sandboxID)
            // Clear sandboxID so cleanup doesn't try to destroy again
            state.sandboxID = ""
        })
    }
    ```
