create_playbook.go skeleton (mirrors create_slack_inbound.go almost line-for-line):
```go
package cmd

type playbookDeps struct {
    SQS                 awspkg.SQSClient
    Config              *config.Config
    UpdateSandboxAttr   func(ctx context.Context, sandboxID, attr string, val any) error
    PutSSMParameter     func(ctx context.Context, name, value string, kmsKeyID string) error
    DeleteSSMParameter  func(ctx context.Context, name string) error
}

func provisionPlaybookQueue(ctx context.Context, deps playbookDeps, sandboxID string) (mainURL, dlqURL string, err error) {
    prefix := deps.Config.GetResourcePrefix()  // existing helper
    mainName := awspkg.PlaybookQueueName(prefix, sandboxID)
    dlqName  := awspkg.PlaybookDLQName(prefix, sandboxID)
    kmsKeyID := deps.Config.GetSandboxesKMSKeyID()  // re-use platform CMK; same as slack-inbound
    mainURL, dlqURL, err = awspkg.CreatePlaybookQueues(ctx, deps.SQS, mainName, dlqName, kmsKeyID)
    if err != nil { return "", "", fmt.Errorf("create playbook queues: %w", err) }

    // Persist both URLs to sandboxes DDB row
    if err := deps.UpdateSandboxAttr(ctx, sandboxID, "playbook_queue_url", mainURL); err != nil {
        // best-effort rollback
        _ = awspkg.DeletePlaybookQueues(ctx, deps.SQS, mainURL, dlqURL)
        return "", "", err
    }
    if err := deps.UpdateSandboxAttr(ctx, sandboxID, "playbook_dlq_url", dlqURL); err != nil {
        _ = awspkg.DeletePlaybookQueues(ctx, deps.SQS, mainURL, dlqURL)
        return "", "", err
    }

    // Write SSM parameter (the SCP-workaround read path the runner uses)
    paramName := "/sandbox/" + sandboxID + "/playbook-queue-url"
    if err := deps.PutSSMParameter(ctx, paramName, mainURL, kmsKeyID); err != nil {
        _ = awspkg.DeletePlaybookQueues(ctx, deps.SQS, mainURL, dlqURL)
        return "", "", err
    }

    return mainURL, dlqURL, nil
}

func rollbackPlaybookQueue(ctx context.Context, deps playbookDeps, mainURL, dlqURL, sandboxID string) error {
    var firstErr error
    if err := awspkg.DeletePlaybookQueues(ctx, deps.SQS, mainURL, dlqURL); err != nil { firstErr = err }
    if err := deps.DeleteSSMParameter(ctx, "/sandbox/"+sandboxID+"/playbook-queue-url"); err != nil &amp;&amp; firstErr == nil { firstErr = err }
    return firstErr
}
```

destroy_playbook.go skeleton (mirrors destroy_slack_inbound.go):
```go
func destroyPlaybookQueue(ctx context.Context, deps playbookDeps, sandboxID string) error {
    // Read both URLs from DDB sandboxes row (lookup pattern from destroy_slack_inbound.go)
    sandboxRow, err := getSandboxRow(ctx, sandboxID)
    if err != nil {
        // log + return nil (idempotent — destroy succeeds even if row gone)
        return nil
    }
    mainURL := sandboxRow.PlaybookQueueURL
    dlqURL  := sandboxRow.PlaybookDLQURL
    if mainURL == "" &amp;&amp; dlqURL == "" {
        return nil // nothing to do — never provisioned
    }
    // Delete both queues (idempotent — pkg/aws.DeletePlaybookQueues tolerates NonExistentQueue)
    if err := awspkg.DeletePlaybookQueues(ctx, deps.SQS, mainURL, dlqURL); err != nil {
        log.Warn().Err(err).Msg("playbook queue deletion failed; continuing destroy")
    }
    // Delete SSM param (idempotent)
    paramName := "/sandbox/" + sandboxID + "/playbook-queue-url"
    if err := deps.DeleteSSMParameter(ctx, paramName); err != nil &amp;&amp; !isParameterNotFound(err) {
        log.Warn().Err(err).Msg("playbook SSM param delete failed")
    }
    // CRITICAL: do NOT touch playbook-runs or playbook-sessions tables — SPEC SC-11: history retained
    return nil
}
```
