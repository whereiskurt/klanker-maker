ContentKey + Apply (pkg/playbook/apply.go):
```go
package playbook

import (
    "context"
    "crypto/sha256"
    "fmt"
)

// ContentKey returns the content-addressed S3 key for a playbook.
// Format: "playbooks/{name}/{sha256-hex}.yaml" — SPEC § S3 layout.
func ContentKey(name string, yamlBytes []byte) string {
    sum := sha256.Sum256(yamlBytes)
    return fmt.Sprintf("playbooks/%s/%x.yaml", name, sum)
}

// Apply uploads the playbook to S3 (content-addressed, idempotent) and writes/updates the DDB row.
// Returns (s3_uri, didUpload, error).
//
// Idempotency: If the content-addressed key already exists in S3, skip the PutObject.
// In all cases, write/update the DDB {prefix}-playbooks row with name, s3_uri, applied_at=now, applied_by=operator.
func Apply(ctx context.Context, s3 S3Client, ddb DDBClient, bucket, table, name, appliedBy string, yamlBytes []byte) (s3URI string, didUpload bool, err error)

// Minimal interfaces for testability:
type S3Client interface {
    HeadObject(ctx context.Context, bucket, key string) (exists bool, err error)
    PutObject(ctx context.Context, bucket, key string, body []byte, contentType string) error
}
type DDBClient interface {
    PutPlaybookRow(ctx context.Context, table string, row PlaybookRow) error
}

type PlaybookRow struct {
    Name      string
    S3URI     string
    AppliedAt string  // RFC3339
    AppliedBy string
}
```
