Two are straightforward (queue exists, DLQ depth zero); the third (`playbook_queue_healthy`) is a reconciliation of the SPEC's `playbook_runner_service_active` per planner finding #9 — operator-side systemd-state observability is impossible without SCP-blocked SSM SendCommand, so the check is renamed to use SQS queue attributes as the liveness proxy.

**Reconciliation note (planner finding #9):** SPEC names the third check `playbook_runner_service_active`. This plan implements it as `playbook_queue_healthy` using SQS attributes (ApproximateNumberOfMessagesNotVisible + LastModifiedTimestamp) as a runner-liveness proxy. RESEARCH §8 details why systemd state is not observable from the operator's machine in this account (org-level SCP blocks SSM SendCommand, the only path to query systemctl status remotely). The reconciliation: same intent (is the runner alive and processing?), different implementation (queue-side proxy instead of sandbox-side direct probe). Documented in the SUMMARY and in docs/playbooks.md (plan 71-10).

func checkPlaybookQueueExists(ctx context.Context, list func(context.Context) ([]playbookRow, error), sqs awspkg.SQSClient) CheckResult {
    if list == nil || sqs == nil {
        return CheckResult{Name: "playbook_queue_exists", Status: CheckSkipped, Message: "deps unavailable"}
    }
    rows, err := list(ctx)
    if err != nil { return CheckResult{..., Status: CheckError, Message: err.Error()} }
    if len(rows) == 0 { return CheckResult{..., Status: CheckOK, Message: "no playbook-enabled sandboxes"} }

    var missing []string
    for _, r := range rows {
        if _, err := sqs.GetQueueAttributes(ctx, &amp;sqstypes.GetQueueAttributesInput{QueueUrl: &amp;r.QueueURL, AttributeNames: []sqstypes.QueueAttributeName{sqstypes.QueueAttributeNameQueueArn}}); err != nil {
            missing = append(missing, r.SandboxID + " (main: " + err.Error() + ")")
        }
        if _, err := sqs.GetQueueAttributes(ctx, &amp;sqstypes.GetQueueAttributesInput{QueueUrl: &amp;r.DLQURL, ...}); err != nil {
            missing = append(missing, r.SandboxID + " (dlq: ...)")
        }
    }
    if len(missing) > 0 { return CheckResult{Name: ..., Status: CheckError, Message: "queues missing: " + strings.Join(missing, ", "), Remediation: "km destroy + km create on the affected sandboxes"} }
    return CheckResult{Name: ..., Status: CheckOK, Message: fmt.Sprintf("%d sandboxes have healthy queues", len(rows))}
}
