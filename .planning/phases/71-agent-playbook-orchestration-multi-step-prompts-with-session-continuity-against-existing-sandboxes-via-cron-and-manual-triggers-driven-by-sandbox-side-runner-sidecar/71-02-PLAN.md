Table schemas (verbatim from SPEC § DynamoDB tables and RESEARCH §6):

```hcl
# {prefix}-playbooks
resource "aws_dynamodb_table" "playbooks" {
  name         = var.table_name
  billing_mode = "PAY_PER_REQUEST"
  hash_key     = "name"
  attribute { name = "name"; type = "S" }
  server_side_encryption { enabled = true }
  point_in_time_recovery { enabled = true }
  tags = var.tags
}

# {prefix}-playbook-sessions
resource "aws_dynamodb_table" "playbook_sessions" {
  name         = var.table_name
  billing_mode = "PAY_PER_REQUEST"
  hash_key     = "playbook"
  range_key    = "sandbox_id"
  attribute { name = "playbook"; type = "S" }
  attribute { name = "sandbox_id"; type = "S" }
  ttl { attribute_name = "ttl_expiry"; enabled = true }
  server_side_encryption { enabled = true }
  point_in_time_recovery { enabled = true }
  tags = var.tags
}

# {prefix}-playbook-runs
resource "aws_dynamodb_table" "playbook_runs" {
  name         = var.table_name
  billing_mode = "PAY_PER_REQUEST"
  hash_key     = "run_id"
  attribute { name = "run_id"; type = "S" }
  attribute { name = "playbook"; type = "S" }
  attribute { name = "started_at"; type = "S" }
  global_secondary_index {
    name            = "playbook-started-index"
    hash_key        = "playbook"
    range_key       = "started_at"
    projection_type = "ALL"
  }
  server_side_encryption { enabled = true }
  point_in_time_recovery { enabled = true }
  tags = var.tags
}
```
