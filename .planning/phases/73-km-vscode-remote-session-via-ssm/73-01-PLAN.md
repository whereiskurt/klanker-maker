       // VSCodeEnabled controls whether the km-vscode systemd unit, the

       func, e.g. after `Parse`):

       ```go
       // DefaultVSCodeEnabled returns whether the profile's effective
       // vscodeEnabled is true. The default is true: only an explicit
       // `cli: { vscodeEnabled: false }` returns false. A nil
       // `Spec.CLI` (no cli: block at all) is treated as default-true.
       //
       // Used by pkg/compiler/userdata.go to gate the template block,
       // and may be used by internal/app/cmd/vscode.go for pre-flight
       // checks (Phase 73).
       func DefaultVSCodeEnabled(p *SandboxProfile) bool {
           if p == nil || p.Spec.CLI == nil {
               return true
           }
           if p.Spec.CLI.VSCodeEnabled == nil {
               return true
           }
           return *p.Spec.CLI.VSCodeEnabled
       }
       ```
