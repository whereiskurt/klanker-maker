<critical_research_findings>
1. **Cleanup runs unconditionally** (per planner brief finding #11) — no profile flag check.
   Missing files are non-errors. This avoids needing to store `vscodeEnabled` in DDB metadata.

2. **Cleanup runs AFTER AWS teardown succeeds** — if AWS destroy fails, the operator can re-run
   `km destroy` and the keys are still on disk for them to use with another sandbox or to
   manually inspect. Pre-AWS cleanup would lose state on retry.

3. **Pitfall 7 mitigation**: use `os.IsNotExist` to make `os.Remove` idempotent.
</critical_research_findings>

    // Phase 73: clean up VS Code Remote-SSH operator state.
    // Idempotent — missing files are non-errors (works for pre-Phase-73 sandboxes).
    if homeDir, herr := os.UserHomeDir(); herr == nil {
        sshConfigPath := filepath.Join(homeDir, ".ssh", "config")
        alias := "km-" + sandboxID
        if rmErr := RemoveHost(sshConfigPath, alias); rmErr != nil {

        for _, name := range []string{sandboxID, sandboxID + ".pub"} {
            keyFile := filepath.Join(keysDir, name)
            if err := os.Remove(keyFile); err != nil && !os.IsNotExist(err) {
                log.Warn().Err(err).Str("file", keyFile).
                    Msg("failed to remove vscode key file (non-fatal)")
            }
        }
    } else {
        log.Warn().Err(herr).Str("sandbox_id", sandboxID).
            Msg("could not locate home dir for vscode cleanup (non-fatal)")
    }
