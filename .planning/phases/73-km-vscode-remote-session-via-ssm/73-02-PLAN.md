    2. **Insert the block.** Add this exact block (note the
       single-quoted heredocs — they prevent shell expansion of
       `$TOKEN` inside the wrapper, which is correct because the
       wrapper script must contain the literal `$TOKEN` reference and
       only resolve it when the bash runs at unit-start time). Use
       Go's raw-string Go template — placement is inside
       `userDataTemplate`'s string literal:

       ```
       {{- if .VSCodeEnabled }}
       # ----------------------------------------------------------------
       # Phase 73: km-vscode — VS Code Web session systemd unit + launcher
       # The unit is NOT enabled (no WantedBy) — `km vscode start` issues
       # an explicit `systemctl restart` that rotates the token each time.
       # ----------------------------------------------------------------
       cat > /etc/systemd/system/km-vscode.service << 'KMVSCODEUNIT'
       [Unit]
       Description=Klanker Maker VS Code Web session
       After=network-online.target
       Wants=network-online.target

       [Service]
       Type=simple
       User=sandbox
       Group=sandbox
       ExecStart=/opt/km/bin/km-vscode-launch
       Restart=on-failure
       RestartSec=2
       # Note: deliberately NOT WantedBy=multi-user.target — explicit start only.
       KMVSCODEUNIT
       chmod 0644 /etc/systemd/system/km-vscode.service

       cat > /opt/km/bin/km-vscode-launch << 'KMVSCODELAUNCH'
       #!/usr/bin/env bash
       # /opt/km/bin/km-vscode-launch — rotate token + exec `code serve-web`.
       # Bind 127.0.0.1 only — SSM port-forward is the security boundary.
       set -euo pipefail
       umask 077
       TOKEN=$(openssl rand -hex 32)
       printf '%s' "$TOKEN" > /etc/km/vscode/token
       chown sandbox:sandbox /etc/km/vscode/token
       chmod 0600 /etc/km/vscode/token
       exec /usr/local/bin/code serve-web \
           --host 127.0.0.1 \
           --port 8443 \
           --connection-token "$TOKEN" \
           --accept-server-license-terms
       KMVSCODELAUNCH
       chmod 0755 /opt/km/bin/km-vscode-launch

       mkdir -p /etc/km/vscode
       chown sandbox:sandbox /etc/km/vscode
       chmod 0700 /etc/km/vscode

       # Reload systemd so the FIRST `km vscode start` finds the unit.
       # The unit is NOT enabled and NOT started here — explicit-start only.
       # AMI-baked sandboxes hit this same path on every cloud-init.
       systemctl daemon-reload
       echo "[km-bootstrap] km-vscode unit + launch wrapper installed (explicit start only)"
       {{- end }}
       ```
