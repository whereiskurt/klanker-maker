<status_check_script>
The combined SSM script for `vscode status` (single round-trip per finding #4):

```bash
echo "=== sshd ==="
systemctl is-active sshd 2>&1 || true
echo "=== authkeys exists ==="
test -f /home/sandbox/.ssh/authorized_keys && echo yes || echo no
echo "=== authkeys content ==="
cat /home/sandbox/.ssh/authorized_keys 2>/dev/null | head -1 || true
```

Output parsing:
- sshd active when output contains `=== sshd ===\nactive`
- authkeys present when output contains `=== authkeys exists ===\nyes`

Error discrimination:
- sshd inactive AND authkeys absent → "VS Code not enabled in this sandbox's profile (set
  spec.cli.vscodeEnabled: true and recreate)" (Phase 73 finding #5: this is the discriminator
  for vscodeEnabled=false vs sshd-genuinely-down).
- sshd active but authkeys absent → "Unexpected state: sshd is running but no authorized_keys
  found. Recreate the sandbox."
- sshd inactive but authkeys present → "sshd is not running on the sandbox. Try `km shell <sb> -- sudo systemctl start sshd`."
- both healthy → "VS Code Remote-SSH ready (sshd active, authorized_keys present)".
</status_check_script>

<connection_block>

```
✓ Updated ~/.ssh/config (Host: km-sb-abc123)
✓ Forwarding localhost:2222 → sandbox:22

In VS Code: F1 → "Remote-SSH: Connect to Host..." → km-sb-abc123
Press Ctrl-C to close the tunnel (sshd keeps running on the sandbox).
```

When `--local-port N` is non-default, replace `2222` with `N`.

    ```go
    func newVSCodeStartCmd(cfg *config.Config, fetcher SandboxFetcher, execFn ShellExecFunc, ssmClient SSMSendAPI) *cobra.Command {
        var localPort int
        cmd := &cobra.Command{
            Use:   "start <sandbox-id>",

    func runVSCodeStart(ctx context.Context, cfg *config.Config, fetcher SandboxFetcher, execFn ShellExecFunc, ssmClient SSMSendAPI, sandboxID string, localPort int) error {
        rec, err := fetcher.FetchSandbox(ctx, cfg, sandboxID)
        if err != nil {
            return fmt.Errorf("fetch sandbox: %w", err)
        }
        instanceID := extractResourceID(rec.Resources, ":instance/")
        if instanceID == "" {
            return fmt.Errorf("no EC2 instance found for sandbox %s", sandboxID)
        }

        home, err := os.UserHomeDir()
        if err != nil {
            return fmt.Errorf("locate home directory: %w", err)
        }
        privPath := filepath.Join(home, ".km", "keys", sandboxID)
        if _, err := os.Stat(privPath); err != nil {
            if os.IsNotExist(err) {
                return fmt.Errorf("private key for %s not found at %s. If you created this sandbox on a different machine, copy the ~/.km/keys/%s* files over", sandboxID, privPath, sandboxID)
            }
            return fmt.Errorf("stat private key: %w", err)
        }

        // Combined SSM check (single round-trip).
        script := strings.Join([]string{
            `echo "=== sshd ==="`,
            `systemctl is-active sshd 2>&1 || true`,
            `echo "=== authkeys exists ==="`,
            `test -f /home/sandbox/.ssh/authorized_keys && echo yes || echo no`,
            `echo "=== authkeys content ==="`,
            `cat /home/sandbox/.ssh/authorized_keys 2>/dev/null | head -1 || true`,
        }, "\n")
        out, err := sendSSMAndWait(ctx, ssmClient, instanceID, script)
        if err != nil {
            return fmt.Errorf("ssm pre-flight check: %w", err)
        }
        sshdActive := strings.Contains(out, "=== sshd ===\nactive")
        authkeysPresent := strings.Contains(out, "=== authkeys exists ===\nyes")
        if !sshdActive && !authkeysPresent {
            return fmt.Errorf("VS Code not enabled in this sandbox's profile (set spec.cli.vscodeEnabled: true and recreate the sandbox)")
        }
        if !authkeysPresent {
            return fmt.Errorf("unexpected state: sshd is running but /home/sandbox/.ssh/authorized_keys is absent — recreate the sandbox")
        }
        if !sshdActive {
            return fmt.Errorf("sshd is not running on the sandbox; try `km shell %s -- sudo systemctl start sshd`", sandboxID)
        }

        // Upsert ~/.ssh/config
        sshConfigPath := filepath.Join(home, ".ssh", "config")
        alias := "km-" + sandboxID
        opts := HostOptions{
            HostName:     "localhost",
            Port:         localPort,
            User:         "sandbox",
            IdentityFile: privPath,
        }
        if err := UpsertHost(sshConfigPath, alias, opts); err != nil {
            return fmt.Errorf("upsert ssh-config: %w", err)
        }

        fmt.Printf("✓ Updated ~/.ssh/config (Host: %s)\n", alias)
        fmt.Printf("✓ Forwarding localhost:%d → sandbox:22\n\n", localPort)
        fmt.Printf("In VS Code: F1 → \"Remote-SSH: Connect to Host...\" → %s\n", alias)
        fmt.Printf("Press Ctrl-C to close the tunnel (sshd keeps running on the sandbox).\n\n")

        // Open the foreground port-forward (blocks until Ctrl-C).
        pfCmd := buildPortForwardCmd(ctx, instanceID, rec.Region, strconv.Itoa(localPort), "22")
        pfCmd.Stdin = os.Stdin
        pfCmd.Stdout = os.Stdout
        pfCmd.Stderr = os.Stderr
        return execFn(pfCmd)
    }

    }
    ```
