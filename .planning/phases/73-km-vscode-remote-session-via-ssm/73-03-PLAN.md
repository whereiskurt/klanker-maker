    **runVSCodeStart skeleton:**

    ```go
    func runVSCodeStart(cobraCmd *cobra.Command, cfg *config.Config, fetcher SandboxFetcher, execFn ShellExecFunc, ssmClient SSMSendAPI, sandboxID, localPort string, noForward bool) error {
        ctx := cobraCmd.Context()
        if ctx == nil { ctx = context.Background() }

        fetcher, execFn, ssmClient, err := resolveVSCodeDeps(ctx, cfg, fetcher, execFn, ssmClient)
        if err != nil { return err }

        rec, err := fetcher.FetchSandbox(ctx, sandboxID)
        if err != nil { return fmt.Errorf("fetch sandbox: %w", err) }

        instanceID, err := extractResourceID(rec.Resources, ":instance/")
        if err != nil { return fmt.Errorf("find EC2 instance: %w", err) }

        // Step 1: restart (rotates the token).
        if _, err := sendSSMAndWait(ctx, ssmClient, instanceID, "systemctl restart km-vscode"); err != nil {
            if strings.Contains(err.Error(), "not found") {
                return fmt.Errorf("VS Code not enabled in this sandbox's profile (sandbox %s) — set spec.cli.vscodeEnabled: true and reprovision; see docs/vscode.md", sandboxID)
            }
            return fmt.Errorf("restart km-vscode: %w", err)
        }

        // Step 2: poll is-active (in-shell, single SSM call, ~10s max).
        pollCmd := "for i in $(seq 1 10); do systemctl is-active --quiet km-vscode && exit 0; sleep 1; done; exit 1"
        if _, err := sendSSMAndWait(ctx, ssmClient, instanceID, pollCmd); err != nil {
            // Step 3: fetch journal for diagnostics.
            journal, _ := sendSSMAndWait(ctx, ssmClient, instanceID, "journalctl -u km-vscode --no-pager -n 20")
            return fmt.Errorf("km-vscode failed to become active within 10s — see journal:\n%s\n\n(hint: see docs/vscode.md troubleshooting; common cause: `code` binary missing — install it via initCommands)", journal)
        }

        // Step 4: read token.
        tokenRaw, err := sendSSMAndWait(ctx, ssmClient, instanceID, "cat /etc/km/vscode/token")
        if err != nil { return fmt.Errorf("read token: %w", err) }
        token := strings.TrimSpace(tokenRaw)

        // Step 5: print connection block.
        fmt.Fprintf(cobraCmd.OutOrStdout(), "\nVS Code ready for %s:\n  URL: http://localhost:%s/?tkn=%s\n", sandboxID, localPort, token)
        if noForward {
            fmt.Fprintln(cobraCmd.OutOrStdout(), "  (--no-forward: tunnel NOT started — open a separate `aws ssm start-session ...` or rerun without --no-forward)")
            return nil
        }
        fmt.Fprintf(cobraCmd.OutOrStdout(), "  Forwarding localhost:%s → sandbox:8443 (Ctrl-C to disconnect; server keeps running)\n\n", localPort)

        // Step 6: foreground port-forward.
        c := buildPortForwardCmd(ctx, instanceID, rec.Region, localPort, "8443")
        c.Stdin = os.Stdin
        c.Stdout = os.Stdout
        c.Stderr = os.Stderr
        return execFn(c)
    }
    ```

    **runVSCodeStatus skeleton (Blocker 4 fix — both SSM calls always run):**

    ```go
    func runVSCodeStatus(cobraCmd *cobra.Command, cfg *config.Config, fetcher SandboxFetcher, ssmClient SSMSendAPI, sandboxID string) error {
        ctx := cobraCmd.Context()
        if ctx == nil { ctx = context.Background() }

        fetcher, _, ssmClient, err := resolveVSCodeDeps(ctx, cfg, fetcher, nil, ssmClient)
        if err != nil { return err }

        rec, err := fetcher.FetchSandbox(ctx, sandboxID)
        if err != nil { return fmt.Errorf("fetch sandbox: %w", err) }

        instanceID, err := extractResourceID(rec.Resources, ":instance/")
        if err != nil { return fmt.Errorf("find EC2 instance: %w", err) }

        // Step 1: is-active. The `|| true` makes the shell exit 0 even when
        // the unit is inactive (systemctl is-active exits 3 in that case).
        // sendSSMAndWait then returns nil error with stdout carrying the
        // literal status word ("active", "inactive", "failed", etc.).
        // This is what makes the trim-comparison meaningful AND lets the
        // unconditional journal call below run for both states.
        out, err := sendSSMAndWait(ctx, ssmClient, instanceID, "systemctl is-active km-vscode || true")
        if err != nil {
            return fmt.Errorf("query is-active: %w", err)
        }
        trimmed := strings.TrimSpace(out)
        var statusErr error
        if trimmed == "active" {
            fmt.Fprintln(cobraCmd.OutOrStdout(), "km-vscode: active")
        } else {
            fmt.Fprintf(cobraCmd.OutOrStdout(), "km-vscode: %s\n", trimmed)
            statusErr = fmt.Errorf("km-vscode is %s on %s", trimmed, sandboxID)
        }

        // Step 2: ALWAYS run journalctl, regardless of state — operators want
        // context whether the unit is active (recent activity) or inactive
        // (last failure, if any).
        journal, jerr := sendSSMAndWait(ctx, ssmClient, instanceID, "journalctl -u km-vscode --no-pager -n 20")
        if jerr != nil {
            // Don't mask the primary status error with a journalctl-fetch failure.
            fmt.Fprintf(cobraCmd.OutOrStdout(), "(journalctl fetch failed: %v)\n", jerr)
        } else {
            fmt.Fprintln(cobraCmd.OutOrStdout(), journal)
        }

        return statusErr
    }
    ```

    Note carefully:
    - Step 1's shell command is `systemctl is-active km-vscode || true`,
      NOT bare `systemctl is-active km-vscode`. The `|| true` makes
      sendSSMAndWait return nil error with stdout carrying `"active"`,
      `"inactive"`, `"failed"`, `"activating"`, etc. This unblocks both
      the trim-comparison AND the unconditional journalctl call below.
    - Step 2 ALWAYS runs (no early return between steps 1 and 2 except
      on a hard SSM-level error).
    - The returned error is non-nil whenever the unit is not exactly
      `"active"` — that's what drives the cobra non-zero exit code that
      `TestVSCodeStatus` asserts.

      first SSM call returns nil error with stdout `"inactive\n"`; second
      returns nil error with sample journal text. (The mock spec was
