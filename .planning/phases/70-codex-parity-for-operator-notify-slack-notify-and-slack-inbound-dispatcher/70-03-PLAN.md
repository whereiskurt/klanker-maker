// installStubBinaries writes minimal /usr/local/bin-style stub scripts that

func installStubBinaries(t *testing.T, dir, sentinelPath string) {

// runHook invokes the extracted hook script with the given event, payload, and
// env vars. Returns sentinel file contents (or empty if file absent) and the
// hook's exit code.
func runHook(t *testing.T, event, payload string, env map[string]string) (sentinel string, exit int) {
    t.Helper()
    tmp := t.TempDir()
    hookPath := filepath.Join(tmp, "km-notify-hook")
    if err := os.WriteFile(hookPath, []byte(extractHookScript(t)), 0755); err != nil {
        t.Fatalf("write hook: %v", err)
    }
    sentinelPath := filepath.Join(tmp, "stub.log")
    installStubBinaries(t, tmp, sentinelPath)

    cmd := exec.Command("bash", hookPath, event)
    cmd.Stdin = strings.NewReader(payload)
    cmd.Env = os.Environ()
    cmd.Env = append(cmd.Env, "PATH="+tmp+":"+os.Getenv("PATH"))
    cmd.Env = append(cmd.Env, "KM_SANDBOX_ID=sb-test70")
    for k, v := range env {
        cmd.Env = append(cmd.Env, k+"="+v)
    }
    out, err := cmd.CombinedOutput()
    _ = out // capture for debugging; not asserted
    if exitErr, ok := err.(*exec.ExitError); ok {
        exit = exitErr.ExitCode()
    } else if err == nil {
        exit = 0
    } else {
        t.Logf("hook run error (may be expected): %v", err)
    }
    data, _ := os.ReadFile(sentinelPath)
    return string(data), exit
}
