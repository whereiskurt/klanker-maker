// extractContext returns 'around' chars of `s` centered on the first occurrence
// of `marker`, or "(not found)" if marker is absent.
func extractContext(s, marker string, around int) string {
    i := strings.Index(s, marker)
    if i < 0 {
        return "(not found)"
    }
    start := i - around/2
    if start < 0 {
        start = 0
    }
    end := i + around
