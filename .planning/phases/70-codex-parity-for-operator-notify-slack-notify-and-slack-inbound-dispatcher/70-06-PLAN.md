// extractCrossAgentBlock returns the substring of userdata between
// `if [ "$DO_SWITCH" -eq 1 ]; then` and the matching `fi`. Used to scope
// assertions to the cross-agent block only.
func extractCrossAgentBlock(ud string) string {
    start := strings.Index(ud, `if [ "$DO_SWITCH" -eq 1 ]; then`)
    if start < 0 {
        return ""
    }
    // Find the matching `fi`. Use a depth counter to handle nested if/fi.
    rest := ud[start:]
    depth := 0
    end := -1
    for i := 0; i < len(rest); i++ {
        if strings.HasPrefix(rest[i:], "if ") && (i == 0 || !isWordContinuation(rest, i)) {
            depth++
        } else if strings.HasPrefix(rest[i:], "fi") && (len(rest) == i+2 || rest[i+2] == '\n' || rest[i+2] == ' ' || rest[i+2] == ';') {
            depth--
            if depth == 0 {
                end = i + 2
                break
            }
        }
    }
    if end < 0 {
        return rest
    }
    return rest[:end]
}
