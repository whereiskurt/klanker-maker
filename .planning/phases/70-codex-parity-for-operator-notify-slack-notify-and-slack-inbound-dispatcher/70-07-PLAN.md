func checkCodexHookConfigPresent(
    ctx context.Context,
    listSandboxes func(context.Context) ([]codexSandboxRef, error),
    runSSMCommand SSMCommandRunner,

    name := "codex_hook_config_present"
    if listSandboxes == nil || runSSMCommand == nil {
        return CheckResult{Name: name, Status: CheckSkipped, Message: "codex check deps not configured"}
    }
    sandboxes, err := listSandboxes(ctx)
    if err != nil {
        return CheckResult{Name: name, Status: CheckWARN, Message: fmt.Sprintf("list codex sandboxes: %v", err)}
    }
    if len(sandboxes) == 0 {
        return CheckResult{Name: name, Status: CheckSkipped, Message: "no sandboxes with spec.cli.agent: codex"}
    }
    var drifted []string
    probeCmd := `cat /home/sandbox/.codex/config.toml 2>/dev/null || echo MISSING`
    for _, sb := range sandboxes {
        out, err := runSSMCommand(ctx, sb.InstanceID, sb.Region, probeCmd)
        if err != nil {
            drifted = append(drifted, fmt.Sprintf("%s (SSM error: %v)", sb.SandboxID, err))
            continue
        }
        if strings.Contains(out, "MISSING") || !strings.Contains(out, "codex_hooks = true") {
            drifted = append(drifted, fmt.Sprintf("%s (missing codex_hooks = true)", sb.SandboxID))
            continue
        }
        if !strings.Contains(out, "/opt/km/bin/km-notify-hook PermissionRequest") {
            drifted = append(drifted, fmt.Sprintf("%s (missing PermissionRequest hook)", sb.SandboxID))
            continue

        if !strings.Contains(out, "/opt/km/bin/km-notify-hook Stop") {
            drifted = append(drifted, fmt.Sprintf("%s (missing Stop hook)", sb.SandboxID))
            continue
        }
    }
    if len(drifted) == 0 {
        return CheckResult{Name: name, Status: CheckOK, Message: fmt.Sprintf("%d codex sandbox(es) healthy", len(sandboxes))}
    }
    return CheckResult{
        Name:    name,
        Status:  CheckWARN,
        Message: fmt.Sprintf("%d/%d codex sandbox(es) drifted", len(drifted), len(sandboxes)),
        Details: drifted,
    }
}

// checkAgentTypeConsistency verifies every km-slack-threads row with agent_type
// set matches the corresponding sandbox profile's spec.cli.agent.
//
// Phase 70 SC-7.
func checkAgentTypeConsistency(
    ctx context.Context,
    scanRows func(context.Context) ([]threadAgentRow, error),
    fetchProfile ProfileFetcher,
) CheckResult {
    name := "agent_type_consistency"
    if scanRows == nil || fetchProfile == nil {
        return CheckResult{Name: name, Status: CheckSkipped, Message: "agent_type check deps not configured"}
    }
    rows, err := scanRows(ctx)
    if err != nil {
        return CheckResult{Name: name, Status: CheckWARN, Message: fmt.Sprintf("scan threads table: %v", err)}
    }
    if len(rows) == 0 {
        return CheckResult{Name: name, Status: CheckSkipped, Message: "no rows with agent_type attribute"}
    }
    // Cache profile lookups so we don't re-fetch for repeated sandbox_ids.
    profileCache := make(map[string]*profile.SandboxProfile)
    var drifted []string
    for _, row := range rows {
        prof, cached := profileCache[row.SandboxID]
        if !cached {
            p, err := fetchProfile(ctx, row.SandboxID)
            if err != nil {
                // Profile fetch failure doesn't necessarily mean drift — could be
                // a deleted sandbox. Skip silently (or list separately).
                continue
            }
            prof = p
            profileCache[row.SandboxID] = prof
        }
        profileAgent := "claude"
        if prof != nil && prof.Spec.CLI != nil && prof.Spec.CLI.Agent == "codex" {
            profileAgent = "codex"
        }
        if row.AgentType != profileAgent {
            drifted = append(drifted, fmt.Sprintf("%s/%s: row=%s profile=%s",
                row.ChannelID, row.ThreadTS, row.AgentType, profileAgent))
        }
    }
    if len(drifted) == 0 {
        return CheckResult{Name: name, Status: CheckOK, Message: fmt.Sprintf("%d row(s) consistent", len(rows))}
    }
    return CheckResult{
        Name:    name,
        Status:  CheckWARN,
        Message: fmt.Sprintf("%d/%d row(s) drifted", len(drifted), len(rows)),
        Details: drifted,
    }
}
