4. **Add section 6.5: Spot interruption poll loop** in the template, between iptables (section 6) and SANDBOX_READY (section 7):
   ```
   {{- if .UseSpot }}
   # 6.5. Spot interruption handler (background)
   {{- if .ArtifactPaths }}
   # Write artifact upload helper script
   cat > /opt/km/bin/km-upload-artifacts << 'UPLOADSCRIPT'
   #!/bin/bash
   # Upload configured artifact paths to S3
   BUCKET="{{ .KMArtifactsBucket }}"
   SANDBOX_ID="{{ .SandboxID }}"
   MAX_SIZE_MB={{ .ArtifactMaxSizeMB }}
   ARTIFACT_PATHS=({{ range .ArtifactPaths }}"{{ . }}" {{ end }})
   for APATH in "${ARTIFACT_PATHS[@]}"; do
     find "$APATH" -type f 2>/dev/null | while read FILE; do
       SIZE_MB=$(( $(stat -f%z "$FILE" 2>/dev/null || stat --printf="%s" "$FILE" 2>/dev/null) / 1048576 ))
       if [ "$MAX_SIZE_MB" -gt 0 ] && [ "$SIZE_MB" -gt "$MAX_SIZE_MB" ]; then
         echo "[km-upload] SKIP $FILE (${SIZE_MB}MB > ${MAX_SIZE_MB}MB limit)"
         continue
       fi
       REL_PATH="${FILE#/}"
       aws s3 cp "$FILE" "s3://${BUCKET}/artifacts/${SANDBOX_ID}/${REL_PATH}" 2>&1 || echo "[km-upload] WARN: failed to upload $FILE"
     done
   done
   UPLOADSCRIPT
   chmod +x /opt/km/bin/km-upload-artifacts
   {{- end }}
   (
     while true; do
       SPOT_ACTION=$(curl -sf \
         -H "X-aws-ec2-metadata-token: $IMDS_TOKEN" \
         "http://169.254.169.254/latest/meta-data/spot/termination-time" 2>/dev/null || echo "")
       if [ -n "$SPOT_ACTION" ]; then
         echo "[km-spot] Spot interruption detected at $SPOT_ACTION"
         {{- if .ArtifactPaths }}
         /opt/km/bin/km-upload-artifacts || true
         {{- end }}
         echo "[km-spot] Shutdown handler complete"
         break
       fi
       sleep 5
     done
   ) &
   {{- end }}
   ```
